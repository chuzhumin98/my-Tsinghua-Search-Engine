134     protected void baseCharacterCheck(final char c, final String parameter)
135     throws IllegalArgumentException {
136         // TODO: Too strict?  UNICODE control characters?
137         if (Character.isISOControl(c) || !Character.isValidCodePoint(c)) {
138             throw new IllegalArgumentException("Contains illegal character 0x" +
139                 Integer.toHexString(c) + ": " + parameter);
140         }
141     }

143     protected String checkHeaderValue(final String value)
144     throws IllegalArgumentException {
145         for (int i = 0; i < value.length(); i++) {
146         	final char c = value.charAt(i);
147         	baseCharacterCheck(c, value);
148         	if (Character.isWhitespace(c)) {
149                 throw new IllegalArgumentException("Contains disallowed white space 0x" +
150                     Integer.toHexString(c) + ": " + value);
151         	}
152         }
153         return value;
154     }

156     protected String checkHeaderLineMimetypeParameter(final String parameter)
157     throws IOException {
158     	StringBuilder sb = new StringBuilder(parameter.length());
159     	boolean wasWhitespace = false;
160         for (int i = 0; i < parameter.length(); i++) {
161         	char c = parameter.charAt(i);
162         	if (Character.isWhitespace(c)) {
163         		// Map all to ' ' and collapse multiples into one.
164         		// TODO: Make sure white space occurs in legal location --
165         		// before parameter or inside quoted-string.
166         		if (wasWhitespace) {
167         			continue;
168         		}
169         		wasWhitespace = true;
170         		c = ' ';
171         	} else {
172         		wasWhitespace = false;
173         		baseCharacterCheck(c, parameter);
174         	}
175         	sb.append(c);
176         }
177         
178         return sb.toString();
179     }

181     protected String createRecordHeader(final String type,
182     		final String url, final String create14DigitDate,
183     		final String mimetype, final URI recordId,
184     		final ANVLRecord xtraHeaders, final long contentLength)
185     throws IOException {
186     	final StringBuilder sb =
187     		new StringBuilder(2048/*A SWAG: TODO: Do analysis.*/);
188     	sb.append(WARC_ID).append(CRLF);
189         sb.append(HEADER_KEY_TYPE).append(COLON_SPACE).append(type).
190             append(CRLF);
191         // Do not write a subject-uri if not one present.
192         if (url != null && url.length() > 0) {
193             sb.append(HEADER_KEY_URI).append(COLON_SPACE).
194                 append(checkHeaderValue(url)).append(CRLF);
195         }
196         sb.append(HEADER_KEY_DATE).append(COLON_SPACE).
197             append(create14DigitDate).append(CRLF);
198         if (xtraHeaders != null) {
199             for (final Iterator i = xtraHeaders.iterator(); i.hasNext();) {
200                 sb.append(i.next()).append(CRLF);
201             }
202         }
203 
204         sb.append(HEADER_KEY_ID).append(COLON_SPACE).append('<').
205             append(recordId.toString()).append('>').append(CRLF);
206         if (contentLength > 0) {
207             sb.append(CONTENT_TYPE).append(COLON_SPACE).append(
208                 checkHeaderLineMimetypeParameter(mimetype)).append(CRLF);
209         }
210         sb.append(CONTENT_LENGTH).append(COLON_SPACE).
211             append(Long.toString(contentLength)).append(CRLF);
212     	
213     	return sb.toString();
214     }

227     protected void writeRecord(final String type, final String url,
228     		final String create14DigitDate, final String mimetype,
229     		final URI recordId, ANVLRecord xtraHeaders,
230             final InputStream contentStream, final long contentLength, boolean enforceLength)
231     throws IOException {
232     	if (!TYPES_LIST.contains(type)) {
233     		throw new IllegalArgumentException("Unknown record type: " + type);
234     	}
235     	if (contentLength == 0 &&
236                 (xtraHeaders == null || xtraHeaders.size() <= 0)) {
237     		throw new IllegalArgumentException("Cannot write record " +
238     		    "of content-length zero and base headers only.");
239     	}
240     	
241     	String header;
242     	try {
243     		header = createRecordHeader(type, url,
244     				create14DigitDate, mimetype, recordId, xtraHeaders,
245     				contentLength);
246 
247     	} catch (IllegalArgumentException e) {
248     		logger.log(Level.SEVERE,"could not write record type: " + type 
249     				+ "for URL: " + url, e);
250     		return;
251     	}    	   	
252 
253         try {
254             preWriteRecordTasks();
255             // TODO: Revisit endcoding of header.
256             write(header.getBytes(WARC_HEADER_ENCODING));
257             
258             if (contentStream != null && contentLength > 0) {
259                 // Write out the header/body separator.
260                 write(CRLF_BYTES); // TODO: should this be written even for zero-length?
261             	copyFrom(contentStream, contentLength, enforceLength);
262             }
263             
264             // Write out the two blank lines at end of all records, per spec
265             write(CRLF_BYTES);
266             write(CRLF_BYTES);
267         } finally {
268             postWriteRecordTasks();
269         }
270     }

303     public URI writeWarcinfoRecord(String filename, final String description)
304         	throws IOException {
305         // Strip .open suffix if present.
306         if (filename.endsWith(WriterPoolMember.OCCUPIED_SUFFIX)) {
307         	filename = filename.substring(0,
308         		filename.length() - WriterPoolMember.OCCUPIED_SUFFIX.length());
309         }
310         
311         ANVLRecord headerrecord = new ANVLRecord(1);
312         headerrecord.addLabelValue(HEADER_KEY_FILENAME, filename);
313         
314         // Ugh, hate doing this but barring larger refactoring per-WARC
315         // 'metadata' is coming back as List<String> (?!?)
316         String blockfields = "";
317         if (this.fileMetadata == null) {
318             // only encountered in unit tests?
319             blockfields = "dummy: value";
320         } else {
321             for (String s :  (List<String>) fileMetadata) {
322                 blockfields += s;
323             }
324         }
325         byte[] warcinfoBody;
326         if (description != null && description.length() > 0) {
327             // reconstitute and add new description
328             ANVLRecord blockrecord = ANVLRecord.load(blockfields);
329             blockrecord.addLabelValue(CONTENT_DESCRIPTION, description);
330             warcinfoBody = blockrecord.toString().getBytes("UTF-8");
331         } else {
332             // just use in already rendered form
333             warcinfoBody = blockfields.getBytes("UTF-8");
334         }
335 
336         URI uri = writeWarcinfoRecord("application/warc-fields", headerrecord,
337             new ByteArrayInputStream(warcinfoBody), warcinfoBody.length);
338         return uri;
339     }
