127         super(name, "Experimental WARCWriter processor (Version 0.17)");
128         Type e = addElementToDefinition(
129                 new SimpleType(ATTR_WRITE_REQUESTS,
130                 "Whether to write 'request' type records. " +
131                 "Default is true.", new Boolean(true)));
132         e.setOverrideable(true);
133         e.setExpertSetting(true);
134         e = addElementToDefinition(
135                 new SimpleType(ATTR_WRITE_METADATA,
136                 "Whether to write 'metadata' type records. " +
137                 "Default is true.", new Boolean(true)));
138         e.setOverrideable(true);
139         e.setExpertSetting(true);
140         e = addElementToDefinition(
141                 new SimpleType(ATTR_WRITE_REVISIT_FOR_IDENTICAL_DIGESTS,
142                 "Whether to write 'revisit' type records when a URI's " +
143                 "history indicates the previous fetch had an identical " +
144                 "content digest. " +
145                 "Default is true.", new Boolean(true)));
146         e.setOverrideable(true);
147         e.setExpertSetting(true);
148         e = addElementToDefinition(
149                 new SimpleType(ATTR_WRITE_REVISIT_FOR_NOT_MODIFIED,
150                 "Whether to write 'revisit' type records when a " +
151                 "304-Not Modified response is received. " +
152                 "Default is true.", new Boolean(true)));
153         e.setOverrideable(true);
154         e.setExpertSetting(true);
155     }

162     /***
163      * Writes a CrawlURI and its associated data to store file.
164      * 
165      * Currently this method understands the following uri types: dns, http, and
166      * https.
167      * 
168      * @param curi CrawlURI to process.
169      * 
170      */
171     protected void innerProcess(CrawlURI curi) {
172         // If failure, or we haven't fetched the resource yet, return
173         if (curi.getFetchStatus() <= 0) {
174             return;
175         }
176         
177         // If no recorded content at all, don't write record. Except FTP, which
178         // can have empty content, since the "headers" don't count as content.
179         String scheme = curi.getUURI().getScheme().toLowerCase();
180         long recordLength = curi.getContentSize();
181         if (recordLength <= 0 && !scheme.equals("ftp")) {
182             // getContentSize() should be > 0 if any material (even just
183             // HTTP headers with zero-length body) is available. 
184             return;
185         }
186         
187         try {
188             if (shouldWrite(curi)) {
189                 write(scheme, curi);
190             } else {
191                 logger.info("This writer does not write out scheme " +
192                         scheme + " content");
193             }
194         } catch (IOException e) {
195             curi.addLocalizedError(this.getName(), e, "WriteRecord: " +
196                 curi.toString());
197             logger.log(Level.SEVERE, "Failed write of Record: " +
198                 curi.toString(), e);
199         }
200     }

202     protected void write(final String lowerCaseScheme, final CrawlURI curi)
203     throws IOException {
204         logger.info("writing warc record for " + curi);
205         WriterPoolMember writer = getPool().borrowFile();
206         long position = writer.getPosition();
207         // See if we need to open a new file because we've exceeed maxBytes.
208         // Call to checkFileSize will open new file if we're at maximum for
209         // current file.
210         writer.checkSize();
211         if (writer.getPosition() != position) {
212             // We just closed the file because it was larger than maxBytes.
213             // Add to the totalBytesWritten the size of the first record
214             // in the file, if any.
215             setTotalBytesWritten(getTotalBytesWritten() +
216             	(writer.getPosition() - position));
217             position = writer.getPosition();
218         }
219         
220         WARCWriter w = (WARCWriter)writer;
221         try {
222             // Write a request, response, and metadata all in the one
223             // 'transaction'.
224             final URI baseid = getRecordID();
225             final String timestamp =
226                 ArchiveUtils.getLog14Date(curi.getLong(A_FETCH_BEGAN_TIME));
227             if (lowerCaseScheme.startsWith("http")) {
228                 writeHttpRecords(w, curi, baseid, timestamp); 
229             } else if (lowerCaseScheme.equals("dns")) {
230                 writeDnsRecords(w, curi, baseid, timestamp);
231             } else if (lowerCaseScheme.equals("ftp")) {
232                 writeFtpRecords(w, curi, baseid, timestamp); 
233             } else {
234                 logger.warning("No handler for scheme " + lowerCaseScheme);
235             }
236         } catch (IOException e) {
237             // Invalidate this file (It gets a '.invalid' suffix).
238             getPool().invalidateFile(writer);
239             // Set the writer to null otherwise the pool accounting
240             // of how many active writers gets skewed if we subsequently
241             // do a returnWriter call on this object in the finally block.
242             writer = null;
243             throw e;
244         } finally {
245             if (writer != null) {
246             	setTotalBytesWritten(getTotalBytesWritten() +
247             	     (writer.getPosition() - position));
248                 getPool().returnFile(writer);
249             }
250         }
251         checkBytesWritten();
252     }

254     private void writeFtpRecords(WARCWriter w, final CrawlURI curi, final URI baseid,
255             final String timestamp) throws IOException {
256         ANVLRecord headers = new ANVLRecord(3);
257         headers.addLabelValue(HEADER_KEY_IP, getHostAddress(curi));
258         String controlConversation = curi.getString(A_FTP_CONTROL_CONVERSATION);
259         URI rid = writeFtpControlConversation(w, timestamp, baseid, curi, headers, controlConversation);
260         
261         if (curi.getContentDigest() != null) {
262             headers.addLabelValue(HEADER_KEY_PAYLOAD_DIGEST,
263                 curi.getContentDigestSchemeString());
264         }
265 
266         if (curi.getHttpRecorder() != null) {
267             if (IdenticalDigestDecideRule.hasIdenticalDigest(curi) && 
268                     ((Boolean)getUncheckedAttribute(curi, 
269                         ATTR_WRITE_REVISIT_FOR_IDENTICAL_DIGESTS))) {
270                 rid = writeRevisitDigest(w, timestamp, null,
271                         baseid, curi, headers);
272             } else {
273                 headers = new ANVLRecord(3);
274                 if (curi.isTruncatedFetch()) {
275                     String value = curi.isTimeTruncatedFetch()?
276                         NAMED_FIELD_TRUNCATED_VALUE_TIME:
277                         curi.isLengthTruncatedFetch()?
278                         NAMED_FIELD_TRUNCATED_VALUE_LENGTH:
279                         curi.isHeaderTruncatedFetch()?
280                         NAMED_FIELD_TRUNCATED_VALUE_HEAD:
281                         // TODO: Add this to spec.
282                         TRUNCATED_VALUE_UNSPECIFIED;
283                     headers.addLabelValue(HEADER_KEY_TRUNCATED, value);
284                 }
285                 if (curi.getContentDigest() != null) {
286                     headers.addLabelValue(HEADER_KEY_PAYLOAD_DIGEST,
287                             curi.getContentDigestSchemeString());
288                 }
289                 headers.addLabelValue(HEADER_KEY_CONCURRENT_TO, '<' + rid.toString() + '>');
290                 rid = writeResource(w, timestamp, curi.getContentType(), baseid, curi, headers);
291             }
292         }
293         if(((Boolean)getUncheckedAttribute(curi, ATTR_WRITE_METADATA))) {
294             headers = new ANVLRecord(1);
295             headers.addLabelValue(HEADER_KEY_CONCURRENT_TO, '<' + rid.toString() + '>');
296             writeMetadata(w, timestamp, baseid, curi, headers);
297         }
298     }

300     private void writeDnsRecords(WARCWriter w, final CrawlURI curi, final URI baseid,
301             final String timestamp) throws IOException {
302         ANVLRecord headers = null;
303         String ip = curi.getString(A_DNS_SERVER_IP_LABEL);
304         if (ip != null && ip.length() > 0) {
305             headers = new ANVLRecord(1);
306             headers.addLabelValue(HEADER_KEY_IP, ip);
307         }
308         writeResponse(w, timestamp, curi.getContentType(), baseid,
309             curi, headers);
310     }

312     private void writeHttpRecords(WARCWriter w, final CrawlURI curi, final URI baseid,
313             final String timestamp) throws IOException {
314         // Add named fields for ip, checksum, and relate the metadata
315         // and request to the resource field.
316         // TODO: Use other than ANVL (or rename ANVL as NameValue or
317         // use RFC822 (commons-httpclient?).
318         ANVLRecord headers = new ANVLRecord(5);
319         if (curi.getContentDigest() != null) {
320             headers.addLabelValue(HEADER_KEY_PAYLOAD_DIGEST,
321                 curi.getContentDigestSchemeString());
322         }
323         headers.addLabelValue(HEADER_KEY_IP, getHostAddress(curi));
324         URI rid;
325         
326         if (IdenticalDigestDecideRule.hasIdenticalDigest(curi) && 
327                 ((Boolean)getUncheckedAttribute(curi, 
328                         ATTR_WRITE_REVISIT_FOR_IDENTICAL_DIGESTS))) {
329             rid = writeRevisitDigest(w, timestamp, HTTP_RESPONSE_MIMETYPE,
330                     baseid, curi, headers);
331         } else if (curi.getFetchStatus() == HttpStatus.SC_NOT_MODIFIED && 
332                 ((Boolean)getUncheckedAttribute(curi, 
333                         ATTR_WRITE_REVISIT_FOR_NOT_MODIFIED))) {
334             rid = writeRevisitNotModified(w, timestamp,
335                     baseid, curi, headers);
336         } else {
337             if (curi.isTruncatedFetch()) {
338                 String value = curi.isTimeTruncatedFetch()?
339                     NAMED_FIELD_TRUNCATED_VALUE_TIME:
340                     curi.isLengthTruncatedFetch()?
341                         NAMED_FIELD_TRUNCATED_VALUE_LENGTH:
342                         curi.isHeaderTruncatedFetch()?
343                             NAMED_FIELD_TRUNCATED_VALUE_HEAD:
344                     // TODO: Add this to spec.
345                     TRUNCATED_VALUE_UNSPECIFIED;
346                 headers.addLabelValue(HEADER_KEY_TRUNCATED, value);
347             }
348             rid = writeResponse(w, timestamp, HTTP_RESPONSE_MIMETYPE,
349             	baseid, curi, headers);
350         }
351         
352         headers = new ANVLRecord(1);
353         headers.addLabelValue(HEADER_KEY_CONCURRENT_TO,
354             '<' + rid.toString() + '>');
355 
356         if(((Boolean)getUncheckedAttribute(curi, ATTR_WRITE_REQUESTS))) {
357             writeRequest(w, timestamp, HTTP_REQUEST_MIMETYPE,
358                     baseid, curi, headers);
359         }
360         if(((Boolean)getUncheckedAttribute(curi, ATTR_WRITE_METADATA))) {
361             writeMetadata(w, timestamp, baseid, curi, headers);
362         }
363     }

375     protected URI writeRequest(final WARCWriter w,
376             final String timestamp, final String mimetype,
377             final URI baseid, final CrawlURI curi,
378             final ANVLRecord namedFields) 
379     throws IOException {
380         final URI uid = qualifyRecordID(baseid, TYPE, REQUEST);
381         ReplayInputStream ris =
382             curi.getHttpRecorder().getRecordedOutput().getReplayInputStream();
383         try {
384             w.writeRequestRecord(curi.toString(), timestamp, mimetype, uid,
385                 namedFields, ris,
386                 curi.getHttpRecorder().getRecordedOutput().getSize());
387         } finally {
388             if (ris != null) {
389                 ris.close();
390             }
391         }
392         return uid;
393     }

433     protected URI writeRevisitDigest(final WARCWriter w,
434             final String timestamp, final String mimetype,
435             final URI baseid, final CrawlURI curi,
436             final ANVLRecord namedFields) 
437     throws IOException {
438         namedFields.addLabelValue(
439                 HEADER_KEY_PROFILE, PROFILE_REVISIT_IDENTICAL_DIGEST);
440         namedFields.addLabelValue(
441                 HEADER_KEY_TRUNCATED, NAMED_FIELD_TRUNCATED_VALUE_LENGTH);
442         
443         ReplayInputStream ris = null;
444         long revisedLength = 0;
445         
446         // null mimetype implies no payload
447         if (mimetype != null) {
448             ris = curi.getHttpRecorder().getRecordedInput().getReplayInputStream();
449             revisedLength = curi.getHttpRecorder().getRecordedInput().getContentBegin();
450             revisedLength = revisedLength > 0 
451                 ? revisedLength 
452                 : curi.getHttpRecorder().getRecordedInput().getSize();
453         }
454         
455         try {
456             w.writeRevisitRecord(curi.toString(), timestamp, mimetype, baseid,
457                 namedFields, ris, revisedLength);
458         } finally {
459             if (ris != null) {
460                 ris.close();
461             }
462         }
463         curi.addAnnotation("warcRevisit:digest"); 
464         return baseid;
465     }

467     protected URI writeRevisitNotModified(final WARCWriter w,
468             final String timestamp, 
469             final URI baseid, final CrawlURI curi,
470             final ANVLRecord namedFields) 
471     throws IOException {
472         namedFields.addLabelValue(
473         		HEADER_KEY_PROFILE, PROFILE_REVISIT_NOT_MODIFIED);
474         // save just enough context to understand basis of not-modified
475         if(curi.containsKey(A_HTTP_TRANSACTION)) {
476             HttpMethodBase method = 
477                 (HttpMethodBase) curi.getObject(A_HTTP_TRANSACTION);
478             saveHeader(A_ETAG_HEADER,method,namedFields,HEADER_KEY_ETAG);
479             saveHeader(A_LAST_MODIFIED_HEADER,method,namedFields,
480             		HEADER_KEY_LAST_MODIFIED);
481         }
482         // truncate to zero-length (all necessary info is above)
483         namedFields.addLabelValue(HEADER_KEY_TRUNCATED,
484             NAMED_FIELD_TRUNCATED_VALUE_LENGTH);
485         ReplayInputStream ris =
486             curi.getHttpRecorder().getRecordedInput().getReplayInputStream();
487         try {
488             w.writeRevisitRecord(curi.toString(), timestamp, null, baseid,
489                 namedFields, ris, 0);
490         } finally {
491             if (ris !=  null) {
492                 ris.close();
493             }
494         }
495         curi.addAnnotation("warcRevisit:notModified"); 
496         return baseid;
497     }

504      * @param method http operation containing headers

514 	protected URI writeMetadata(final WARCWriter w,
515             final String timestamp,
516             final URI baseid, final CrawlURI curi,
517             final ANVLRecord namedFields) 
518     throws IOException {
519         final URI uid = qualifyRecordID(baseid, TYPE, METADATA);
520         // Get some metadata from the curi.
521         // TODO: Get all curi metadata.
522         // TODO: Use other than ANVL (or rename ANVL as NameValue or use
523         // RFC822 (commons-httpclient?).
524         ANVLRecord r = new ANVLRecord();
525         if (curi.isSeed()) {
526             r.addLabel("seed");
527         } else {
528         	if (curi.forceFetch()) {
529         		r.addLabel("force-fetch");
530         	}
531             r.addLabelValue("via", curi.flattenVia());
532             r.addLabelValue("hopsFromSeed", curi.getPathFromSeed());
533             if (curi.containsKey(A_SOURCE_TAG)) {
534                 r.addLabelValue("sourceTag", curi.getString(A_SOURCE_TAG));
535             }
536         }
537         long duration = curi.getFetchDuration();
538         if(duration>-1) {
539             r.addLabelValue("fetchTimeMs", Long.toString(duration));
540         }
541         
542         if (curi.containsKey(A_FTP_FETCH_STATUS)) {
543             r.addLabelValue("ftpFetchStatus", curi.getString(A_FTP_FETCH_STATUS));
544         }
545         
546         // Add outlinks though they are effectively useless without anchor text.
547         Collection<Link> links = curi.getOutLinks();
548         if (links != null && links.size() > 0) {
549             for (Link link: links) {
550                 r.addLabelValue("outlink", link.toString());
551             }
552         }
553         
554         // TODO: Other curi fields to write to metadata.
555         // 
556         // Credentials
557         // 
558         // fetch-began-time: 1154569278774
559         // fetch-completed-time: 1154569281816
560         //
561         // Annotations.
562         
563         byte [] b = r.getUTF8Bytes();
564         w.writeMetadataRecord(curi.toString(), timestamp, ANVLRecord.MIMETYPE,
565             uid, namedFields, new ByteArrayInputStream(b), b.length);
566         return uid;
567     }

599     /***
600      * Return relevant values as header-like fields (here ANVLRecord, but 
601      * spec-defined "application/warc-fields" type when written). Field
602      * names from from DCMI Terms and the WARC/0.17 specification.
603      * 
604      * @see org.archive.crawler.framework.WriterPoolProcessor#getFirstrecordBody(java.io.File)
605      */
606     @Override
607     protected String getFirstrecordBody(File orderFile) {
608         ANVLRecord record = new ANVLRecord(7);
609         record.addLabelValue("software", "Heritrix/" +
610                 Heritrix.getVersion() + " http://crawler.archive.org");
611         try {
612             InetAddress host = InetAddress.getLocalHost();
613             record.addLabelValue("ip", host.getHostAddress());
614             record.addLabelValue("hostname", host.getCanonicalHostName());
615         } catch (UnknownHostException e) {
616             logger.log(Level.WARNING,"unable top obtain local crawl engine host",e);
617         }
618 
619         // conforms to ISO 28500:2009 as of May 2009
620         // as described at http://bibnum.bnf.fr/WARC/ 
621         // latest draft as of November 2008
622         record.addLabelValue("format","WARC File Format 1.0"); 
623         record.addLabelValue("conformsTo","http://bibnum.bnf.fr/WARC/WARC_ISO_28500_version1_latestdraft.pdf");
624 
625         // Get other values from order.xml 
626         try {
