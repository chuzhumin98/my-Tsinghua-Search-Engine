107     /***
108      * This methods updates a ComplexType with information passed to it by a
109      * HttpServletRequest. It assumes that for every 'simple' type there is a
110      * corresponding parameter in the request. A recursive call will be made for
111      * any nested ComplexTypes. For each attribute it will check if the relevant
112      * override is set (name.override parameter equals 'true'). If so the
113      * attribute setting on the specified domain level (settings) will be
114      * rewritten. If it is not we well ensure that it isn't being overridden.
115      * 
116      * @param mbean
117      *            The ComplexType to update
118      * @param settings
119      *            CrawlerSettings for the domain to override setting for. null
120      *            denotes the global settings.
121      * @param request
122      *            The HttpServletRequest to use to update the ComplexType
123      * @param expert
124      *            if true expert settings will be updated, otherwise they will
125      *            be ignored.
126      */
127     public static void writeNewOrderFile(ComplexType mbean,
128             CrawlerSettings settings, HttpServletRequest request, boolean expert) {
129         // If mbean is transient or a hidden expert setting.
130         if (mbean.isTransient() || (mbean.isExpertSetting() && expert == false)) {
131             return;
132         }
133 
134         MBeanAttributeInfo a[] = mbean.getMBeanInfo(settings).getAttributes();
135         for (int n = 0; n < a.length; n++) {
136             checkAttribute((ModuleAttributeInfo) a[n], mbean, settings,
137                     request, expert);
138         }
139     }

141     /***
142      * Process passed attribute. Check if needs to be written and if so, write
143      * it.
144      * 
145      * @param att
146      *            Attribute to process.
147      * @param mbean
148      *            The ComplexType to update
149      * @param settings
150      *            CrawlerSettings for the domain to override setting for. null
151      *            denotes the global settings.
152      * @param request
153      *            The HttpServletRequest to use to update the ComplexType
154      * @param expert
155      *            if true expert settings will be updated, otherwise they will
156      *            be ignored.
157      */
158     @SuppressWarnings("unchecked")
159     protected static void checkAttribute(ModuleAttributeInfo att,
160             ComplexType mbean, CrawlerSettings settings,
161             HttpServletRequest request, boolean expert) {
162         // The attributes of the current attribute.
163         Object currentAttribute = null;
164         try {
165             currentAttribute = mbean.getAttribute(settings, att.getName());
166         } catch (Exception e) {
167             logger.severe("Failed getting " + mbean.getAbsoluteName()
168                     + " attribute " + att.getName() + ": " + e.getMessage());
169             return;
170         }
171 
172         if (logger.isLoggable(Level.FINE)) {
173             logger.fine("MBEAN: " + mbean.getAbsoluteName() + " "
174                     + att.getName() + " TRANSIENT " + att.isTransient() + " "
175                     + att.isExpertSetting() + " " + expert);
176         }
177 
178         if (att.isTransient() == false
179                 && (att.isExpertSetting() == false || expert)) {
180             if (currentAttribute instanceof ComplexType) {
181                 writeNewOrderFile((ComplexType) currentAttribute, settings,
182                         request, expert);
183             } else {
184                 String attName = att.getName();
185                 // Have a 'setting'. Let's see if we need to update it (if
186                 // settings == null update all, otherwise only if override
187                 // is set.
188                 String attAbsoluteName = mbean.getAbsoluteName() + "/"
189                         + attName;
190                 boolean override = (request.getParameter(attAbsoluteName
191                         + ".override") != null)
192                         && (request.getParameter(attAbsoluteName + ".override")
193                                 .equals("true"));
194                 if (settings == null || override) {
195                     if (currentAttribute instanceof ListType) {
196                         try {
197                             ListType list = (ListType)currentAttribute;
198                             Class cls = list.getClass();
199                             Constructor constructor = cls.getConstructor(String.class, String.class);
200                             list = (ListType) constructor.newInstance(list.getName(), list.getDescription());
201                             String[] elems = request
202                                     .getParameterValues(attAbsoluteName);
203                             for (int i = 0; elems != null && i < elems.length; i++) {
204                                 list.add(elems[i]);
205                             }
206                             writeAttribute(attName, attAbsoluteName, mbean,
207                                     settings, list);
208                         } catch (Exception e) {
209                             e.printStackTrace();
210                             logger.severe("Setting new list values on "
211                                     + attAbsoluteName + ": " + e.getMessage());
212                             return;
213                         }
214                     } else {
215                         writeAttribute(attName, attAbsoluteName, mbean,
216                                 settings, request.getParameter(attAbsoluteName));
217                     }
218 
219                 } else if (settings != null && override == false) {
220                     // Is not being overridden. Need to remove possible
221                     // previous overrides.
222                     try {
223                         mbean.unsetAttribute(settings, attName);
224                     } catch (Exception e) {
225                         e.printStackTrace();
226                         logger.severe("Unsetting attribute on "
227                                 + attAbsoluteName + ": " + e.getMessage());
228                         return;
229                     }
230                 }
231             }
232         }
233     }

235     /***
236      * Write out attribute.
237      * 
238      * @param attName
239      *            Attribute short name.
240      * @param attAbsoluteName
241      *            Attribute full name.
242      * @param mbean
243      *            The ComplexType to update
244      * @param settings
245      *            CrawlerSettings for the domain to override setting for. null
246      *            denotes the global settings.
247      * @param value
248      *            Value to set into the attribute.
249      */
250     protected static void writeAttribute(String attName,
251             String attAbsoluteName, ComplexType mbean,
252             CrawlerSettings settings, Object value) {
253         try {
254             if (logger.isLoggable(Level.FINE)) {
255                 logger.fine("MBEAN SET: " + attAbsoluteName + " " + value);
256             }
257             mbean.setAttribute(settings, new Attribute(attName, value));
258         } catch (Exception e) {
259             e.printStackTrace();
260             logger.severe("Setting attribute value " + value + " on "
261                     + attAbsoluteName + ": " + e.getMessage());
262             return;
263         }
264     }

266     /***
267      * Check passed job is not null and not readonly.
268      * @param job Job to check.
269      * @param response Http response.
270      * @param redirectBasePath Full path for where to go next if an error.
271      * @param currDomain May be null.
272      * E.g. "/admin/jobs/per/overview.jsp".
273      * @return A job else we've redirected if no job or readonly.
274      * @throws IOException
275      */
276     public static CrawlJob checkCrawlJob(CrawlJob job,
277         HttpServletResponse response, String redirectBasePath,
278         String currDomain)
279     throws IOException {
280         if (job == null) {
281             // Didn't find any job with the given UID or no UID given.
282             response.sendRedirect(redirectBasePath +
283                 "?message=No job selected");
284         } else if (job.isReadOnly()) {
285             // Can't edit this job.
286             response.sendRedirect(redirectBasePath +
287                 "?job=" + job.getUID() +
288                 ((currDomain != null && currDomain.length() > 0)?
289                     "&currDomain=" + currDomain: "") +
290                 "&message=Can't edit a read only job");
291         }
292         return job;
293     }

295     /***
296      * Handle job action.
297      * @param handler CrawlJobHandler to operate on.
298      * @param request Http request.
299      * @param response Http response.
300      * @param redirectBasePath Full path for where to go next if an error.
301      * E.g. "/admin/jobs/per/overview.jsp".
302      * @param currDomain Current domain.  Pass null for global domain.
303      * @param reference 
304      * @return The crawljob configured.
305      * @throws IOException
306      * @throws AttributeNotFoundException
307      * @throws InvocationTargetException
308      * @throws InvalidAttributeValueException
309      */
310     public static CrawlJob handleJobAction(CrawlJobHandler handler,
311             HttpServletRequest request, HttpServletResponse response,
312             String redirectBasePath, String currDomain, String reference)
313     throws IOException, AttributeNotFoundException, InvocationTargetException,
314         InvalidAttributeValueException {
315 
316         // Load the job to manipulate
317         CrawlJob theJob =
318             checkCrawlJob(handler.getJob(request.getParameter("job")),
319                 response, redirectBasePath, currDomain);
320 
321         XMLSettingsHandler settingsHandler = theJob.getSettingsHandler();
322         // If currDomain is null, then we're at top-level.
323         CrawlerSettings settings = settingsHandler
324             .getSettingsObject(currDomain);
325         
326         if(reference != null) {
327             // refinement
328             Refinement refinement = settings.getRefinement(reference);
329             settings = refinement.getSettings();
330         }
331 
332         // See if we need to take any action
333         if (request.getParameter(ACTION) != null) {
334             // Need to take some action.
335             String action = request.getParameter(ACTION);
336             String subaction = request.getParameter(SUBACTION);
337             if (action.equals(FILTERS)) {
338                 // Doing something with the filters.
339                 String map = request.getParameter(MAP);
340                 if (map != null && map.length() > 0) {
341                     String filter = request.getParameter(FILTER);
342                     MapType filterMap = (MapType) settingsHandler
343                         .getComplexTypeByAbsoluteName(settings, map);
344                     if (subaction.equals(ADD)) {
345                         // Add filter
346                         String className = request.getParameter(map + ".class");
347                         String typeName = request.getParameter(map + ".name");
348                         if (typeName != null && typeName.length() > 0 &&
349                                 className != null && className.length() > 0) {
350                             ModuleType tmp = SettingsHandler
351                                 .instantiateModuleTypeFromClassName(
352                                     typeName, className);
353                             filterMap.addElement(settings, tmp);
354                         }
355                     } else if (subaction.equals(MOVEUP)) {
356                         // Move a filter down in a map
357                         if (filter != null && filter.length() > 0) {
358                             filterMap.moveElementUp(settings, filter);
359                         }
360                     } else if (subaction.equals(MOVEDOWN)) {
361                         // Move a filter up in a map
362                         if (filter != null && filter.length() > 0) {
363                             filterMap.moveElementDown(settings, filter);
364                         }
365                     } else if (subaction.equals(REMOVE)) {
366                         // Remove a filter from a map
367                         if (filter != null && filter.length() > 0) {
368                             filterMap.removeElement(settings, filter);
369                         }
370                     }
371                 }
372                 // Finally save the changes to disk
373                 settingsHandler.writeSettingsObject(settings);
374             } else if (action.equals(DONE)) {
375                 // Ok, done editing.
376                 if(subaction.equals(CONTINUE)) {
377                     // was editting an override/refinement, simply continue
378                     if (theJob.isRunning()) {
379                         handler.kickUpdate(); //Just to make sure.
380                     }
381                     String overParam = ((currDomain != null && currDomain
382                             .length() > 0) ? "&currDomain=" + currDomain : "");
383                     String refParam = 
384                         ((reference != null && reference.length() > 0) 
385                                 ? "&reference=" + reference
386                                 : "");
387                     String messageParam = (refParam.length() > 0) 
388                          ? "&message=Refinement changes saved"
389                          : "&message=Override changes saved";
390                     response.sendRedirect(redirectBasePath +
391                         "?job=" + theJob.getUID() +
392                          overParam +
393                          refParam + 
394                          messageParam);
395                 } else {
396                     // on main, truly 'done'
397                     if (theJob.isNew()) {
398                         handler.addJob(theJob);
399                         response.sendRedirect(redirectBasePath
400                                 + "?message=Job created");
401                     } else {
402                         if (theJob.isRunning()) {
403                             handler.kickUpdate();
404                         }
405                         if (theJob.isProfile()) {
406                             response.sendRedirect(redirectBasePath
407                                     + "?message=Profile modified");
408                         } else {
409                             response.sendRedirect(redirectBasePath
410                                     + "?message=Job modified");
411                         }
412                     }
413                 }
414             } else if (action.equals(GOTO)) {
415                 // Goto another page of the job/profile settings
416                 String overParam = ((currDomain != null && currDomain
417                         .length() > 0) ? "&currDomain=" + currDomain : "");
418                 String refParam = 
419                     ((reference != null && reference.length() > 0) 
420                             ? "&reference=" + reference
421                             : "");
422                 response.sendRedirect(request.getParameter(SUBACTION) +
423                     overParam + refParam);
424             }
425         }
426         return theJob;
427     }

429     /***
430      * Print complete seeds list on passed in PrintWriter.
431      * @param hndlr Current handler.
432      * @param payload What to write out.
433      * @throws AttributeNotFoundException
434      * @throws MBeanException
435      * @throws ReflectionException
436      * @throws IOException
437      * @throws IOException
438      */
439     public static void printOutSeeds(SettingsHandler hndlr, String payload)
440     throws AttributeNotFoundException, MBeanException, ReflectionException,
441     IOException {
442         File seedfile = getSeedFile(hndlr);
443         // no matter the encoding of the returned page, or the encoding
444         // set on the JSP request before getParameter, the Strings we 
445         // get back are UTF-8-bytes-as-if-ISO8859-1... so reinterpret
446         String utf8 = new String(payload.getBytes("ISO8859-1"),"UTF-8");
447         Writer out = new OutputStreamWriter(new FileOutputStream(seedfile, false),"UTF-8");
448         IOUtils.copy(
449             new StringReader(utf8), 
450             out);
451         out.close();
452     }

463     public static void printOutSeeds(SettingsHandler hndlr, Writer out)

465             IOException {
466         // getSeedStream looks for seeds on disk and on classpath.

500     /***
501      * Return seeds as a stream.
502      * This method will work for case where seeds are on disk or on classpath.
503      * @param hndlr SettingsHandler.  Used to find seeds.txt file.
504      * @return InputStream on current seeds file.
505      * @throws IOException
506      * @throws ReflectionException
507      * @throws MBeanException
508      * @throws AttributeNotFoundException
509      */
510     protected static InputStream getSeedStream(SettingsHandler hndlr)
511     throws IOException, AttributeNotFoundException, MBeanException,
512             ReflectionException {
513         InputStream is = null;
514         File seedFile = getSeedFile(hndlr);
515         if (!seedFile.exists()) {
516             // Is the file on the CLASSPATH?
517             is = SettingsHandler.class.
518                 getResourceAsStream(IoUtils.getClasspathPath(seedFile));
519         } else if(seedFile.canRead()) {
520             is = new FileInputStream(seedFile);
521         }
522         if (is == null) {
523             throw new IOException(seedFile + " does not" +
524             " exist -- neither on disk nor on CLASSPATH -- or is not" +
525             " readable.");
526         }
527         return is;
528     }
