65  /***
66   * This is an implementation of the AbstractTracker. It is designed to function
67   * with the WUI as well as performing various logging activity.
68   * <p>
69   * At the end of each snapshot a line is written to the
70   * 'progress-statistics.log' file.
71   * <p>
72   * The header of that file is as follows:
73   * <pre> [timestamp] [discovered]    [queued] [downloaded] [doc/s(avg)]  [KB/s(avg)] [dl-failures] [busy-thread] [mem-use-KB]</pre>
74   * First there is a <b>timestamp</b>, accurate down to 1 second.
75   * <p>
76   * <b>discovered</b>, <b>queued</b>, <b>downloaded</b> and <b>dl-failures</b>
77   * are (respectively) the discovered URI count, pending URI count, successfully
78   * fetched count and failed fetch count from the frontier at the time of the
79   * snapshot.
80   * <p>
81   * <b>KB/s(avg)</b> is the bandwidth usage.  We use the total bytes downloaded
82   * to calculate average bandwidth usage (KB/sec). Since we also note the value
83   * each time a snapshot is made we can calculate the average bandwidth usage
84   * during the last snapshot period to gain a "current" rate. The first number is
85   * the current and the average is in parenthesis.
86   * <p>
87   * <b>doc/s(avg)</b> works the same way as doc/s except it show the number of
88   * documents (URIs) rather then KB downloaded.
89   * <p>
90   * <b>busy-threads</b> is the total number of ToeThreads that are not available
91   * (and thus presumably busy processing a URI). This information is extracted
92   * from the crawl controller.
93   * <p>
94   * Finally mem-use-KB is extracted from the run time environment
95   * (<code>Runtime.getRuntime().totalMemory()</code>).
96   * <p>
97   * In addition to the data collected for the above logs, various other data
98   * is gathered and stored by this tracker.
99   * <ul>
100  *   <li> Successfully downloaded documents per fetch status code
101  *   <li> Successfully downloaded documents per document mime type
102  *   <li> Amount of data per mime type
103  *   <li> Successfully downloaded documents per host
104  *   <li> Amount of data per host
105  *   <li> Disposition of all seeds (this is written to 'reports.log' at end of
106  *        crawl)
107  *   <li> Successfully downloaded documents per host per source
108  * </ul>
109  *
110  * @author Parker Thompson
111  * @author Kristinn Sigurdsson
112  * 
113  * @see org.archive.crawler.framework.StatisticsTracking
114  * @see org.archive.crawler.framework.AbstractTracker
115  */

212         try {
213             this.sourceHostDistribution = c.getBigMap("sourceHostDistribution",
214             	ConcurrentMap.class);
215             this.hostsDistribution = c.getBigMap("hostsDistribution",
216                 AtomicLong.class);
217             this.hostsBytes = c.getBigMap("hostsBytes", AtomicLong.class);
218             this.hostsLastFinished = c.getBigMap("hostsLastFinished",

254     protected synchronized void progressStatisticsEvent(final EventObject e) {
255         // This method loads "snapshot" data.
256         discoveredUriCount = discoveredUriCount();
257         downloadedUriCount = successfullyFetchedCount();
258         finishedUriCount = finishedUriCount();
259         queuedUriCount = queuedUriCount();
260         downloadFailures = failedFetchAttempts();
261         downloadDisregards = disregardedFetchAttempts();
262         totalProcessedBytes = totalBytesCrawled();
263         congestionRatio = congestionRatio();
264         deepestUri = deepestUri();
265         averageDepth = averageDepth();
266         
267         if (finishedUriCount() == 0) {
268             docsPerSecond = 0;
269             totalKBPerSec = 0;
270         } else if (getCrawlerTotalElapsedTime() < 1000) {
271             return; // Not enough time has passed for a decent snapshot.
272         } else {
273             docsPerSecond = (double) downloadedUriCount /
274                 (double)(getCrawlerTotalElapsedTime() / 1000);
275             // Round to nearest long.
276             totalKBPerSec = (long)(((totalProcessedBytes / 1024) /
277                  ((getCrawlerTotalElapsedTime()) / 1000)) + .5 );
278         }
279 
280         busyThreads = activeThreadCount();
281 
282         if(shouldrun ||
283             (System.currentTimeMillis() - lastLogPointTime) >= 1000) {
284             // If shouldrun is false there is a chance that the time interval
285             // since last time is too small for a good sample.  We only want
286             // to update "current" data when the interval is long enough or
287             // shouldrun is true.
288             currentDocsPerSecond = 0;
289             currentKBPerSec = 0;
290 
291             // Note time.
292             long currentTime = System.currentTimeMillis();
293             long sampleTime = currentTime - lastLogPointTime;
294 
295             // if we haven't done anyting or there isn't a reasonable sample
296             // size give up.
297             if (sampleTime >= 1000) {
298                 // Update docs/sec snapshot
299                 long currentPageCount = successfullyFetchedCount();
300                 long samplePageCount = currentPageCount - lastPagesFetchedCount;
301 
302                 currentDocsPerSecond =
303                     (double) samplePageCount / (double)(sampleTime / 1000);
304 
305                 lastPagesFetchedCount = currentPageCount;
306 
307                 // Update kbytes/sec snapshot
308                 long currentProcessedBytes = totalProcessedBytes;
309                 long sampleProcessedBytes =
310                     currentProcessedBytes - lastProcessedBytesCount;
311 
312                 currentKBPerSec =
313                     (int)(((sampleProcessedBytes/1024)/(sampleTime/1000)) + .5);
314 
315                 lastProcessedBytesCount = currentProcessedBytes;
316             }
317         }
318 
319         if (this.controller != null) {
320             this.controller.logProgressStatistics(getProgressStatisticsLine());
321         }
322         lastLogPointTime = System.currentTimeMillis();
323         super.progressStatisticsEvent(e);
324     }

326     /***
327      * Return one line of current progress-statistics
328      * 
329      * @param now
330      * @return String of stats
331      */
332     public String getProgressStatisticsLine(Date now) {
333         return new PaddingStringBuffer()
334             .append(ArchiveUtils.getLog14Date(now))
335             .raAppend(32, discoveredUriCount)
336             .raAppend(44, queuedUriCount)
337             .raAppend(57, downloadedUriCount)
338             .raAppend(74, ArchiveUtils.
339                 doubleToString(currentDocsPerSecond, 2) +
340                 "(" + ArchiveUtils.doubleToString(docsPerSecond, 2) + ")")
341             .raAppend(85, currentKBPerSec + "(" + totalKBPerSec + ")")
342             .raAppend(99, downloadFailures)
343             .raAppend(113, busyThreads)
344             .raAppend(126, (Runtime.getRuntime().totalMemory() -
345                 Runtime.getRuntime().freeMemory()) / 1024)
346             .raAppend(140, Runtime.getRuntime().totalMemory() / 1024)
347             .raAppend(153, ArchiveUtils.doubleToString(congestionRatio, 2))
348             .raAppend(165, deepestUri)
349             .raAppend(177, averageDepth)
350             .toString();
351     }

462         lw.addAndGet(increment);

465     /***
466      * Increment a counter for a key in a given HashMap by an arbitrary amount.
467      * Used for various aggregate data. The increment amount can be negative.
468      *
469      * @param map
470      *            The Map or ConcurrentMap
471      * @param key
472      *            The key for the counter to be incremented, if it does not exist
473      *            it will be added (set to equal to <code>increment</code>).
474      *            If null it will increment the counter "unknown".
475      * @param increment
476      *            The amount to increment counter related to the <code>key</code>.
477      */
478     protected static void incrementMapCount(ConcurrentMap<String,AtomicLong> map, 
479             String key, long increment) {
480         if (key == null) {
481             key = "unknown";
482         }
483         AtomicLong lw = map.get(key);
484         if(lw == null) {
485             lw = new AtomicLong();
486             AtomicLong prevVal = map.putIfAbsent(key, lw);
487             if(prevVal != null) {
488                 lw = prevVal;
489             }
490         } 
491         lw.addAndGet(increment);
492     }

494     /***
495      * Sort the entries of the given HashMap in descending order by their
496      * values, which must be longs wrapped with <code>AtomicLong</code>.
497      * <p>
498      * Elements are sorted by value from largest to smallest. Equal values are
499      * sorted in an arbitrary, but consistent manner by their keys. Only items
500      * with identical value and key are considered equal.
501      *
502      * If the passed-in map requires access to be synchronized, the caller
503      * should ensure this synchronization. 
504      * 
505      * @param mapOfAtomicLongValues
506      *            Assumes values are wrapped with AtomicLong.
507      * @return a sorted set containing the same elements as the map.
508      */
509     public TreeMap<String,AtomicLong> getReverseSortedCopy(
510             final Map<String,AtomicLong> mapOfAtomicLongValues) {
511         TreeMap<String,AtomicLong> sortedMap = 
512           new TreeMap<String,AtomicLong>(new Comparator<String>() {
513             public int compare(String e1, String e2) {
514                 long firstVal = mapOfAtomicLongValues.get(e1).get();
515                 long secondVal = mapOfAtomicLongValues.get(e2).get();
516                 if (firstVal < secondVal) {
517                     return 1;
518                 }
519                 if (secondVal < firstVal) {
520                     return -1;
521                 }
522                 // If the values are the same, sort by keys.
523                 return e1.compareTo(e2);
524             }
525         });
526         try {
527             sortedMap.putAll(mapOfAtomicLongValues);
528         } catch (UnsupportedOperationException e) {
529             for (String key: mapOfAtomicLongValues.keySet()) {
530                 sortedMap.put(key, mapOfAtomicLongValues.get(key));
531             }
532         }
533         return sortedMap;
534     }

639      * This returns the number of completed URIs as a percentage of the total
640      * number of URIs encountered (should be inverse to the discovery curve)
641      *
642      * @return The number of completed URIs as a percentage of the total

645     public int percentOfDiscoveredUrisCompleted() {
646         long completed = finishedUriCount();

651         }
652

823     public void crawledURISuccessful(CrawlURI curi) {
824         handleSeed(curi,SEED_DISPOSITION_SUCCESS);
825         // save crawled bytes tally
826         crawledBytes.accumulate(curi);
827 
828         // save crawled docs tally
829         if(curi.getFetchStatus()==HttpStatus.SC_NOT_MODIFIED) {
830             notModifiedUriCount++;
831         } else if (IdenticalDigestDecideRule.hasIdenticalDigest(curi)) {
832             dupByHashUriCount++;
833         } else {
834             novelUriCount++;
835         }
836         
837         // Save status codes
838         incrementMapCount(statusCodeDistribution,
839             Integer.toString(curi.getFetchStatus()));
840 
841         // Save mime types
842         String mime = MimetypeUtils.truncate(curi.getContentType());
843         incrementMapCount(mimeTypeDistribution, mime);
844         incrementMapCount(mimeTypeBytes, mime, curi.getContentSize());
845 
846         // Save hosts stats.
847         saveHostStats(curi.getFetchStatus() == FetchStatusCodes.S_DNS_SUCCESS ? "dns:" :
848                 this.controller.getServerCache().getHostFor(curi).getHostName(),
849                 curi.getContentSize());
850         
851         if (curi.containsKey(CrawlURI.A_SOURCE_TAG)){
852             saveSourceStats(curi.getString(CrawlURI.A_SOURCE_TAG), 
853                     this.controller.getServerCache().getHostFor(curi).
854                     getHostName()); 
855         }
856     }

914     protected Iterator<SeedRecord> getSeedRecordsSortedByStatusCode(
915             Iterator<String> i) {
916         TreeSet<SeedRecord> sortedSet = 
917           new TreeSet<SeedRecord>(new Comparator<SeedRecord>() {
918             public int compare(SeedRecord sr1, SeedRecord sr2) {
919                 int code1 = sr1.getStatusCode();
920                 int code2 = sr2.getStatusCode();
921                 if (code1 == code2) {
922                     // If the values are equal, sort by URIs.
923                     return sr1.getUri().compareTo(sr2.getUri());
924                 }
925                 // mirror and shift the nubmer line so as to
926                 // place zero at the beginning, then all negatives 
927                 // in order of ascending absolute value, then all 
928                 // positives descending
929                 code1 = -code1 - Integer.MAX_VALUE;
930                 code2 = -code2 - Integer.MAX_VALUE;
931                 
932                 return new Integer(code1).compareTo(new Integer(code2));
933             }
934         });
935         while (i.hasNext()) {
936             String seed = i.next();
937             SeedRecord sr = (SeedRecord) processedSeedsRecords.get(seed);
938             if(sr==null) {
939                 sr = new SeedRecord(seed,SEED_DISPOSITION_NOT_PROCESSED);
940             }
941             sortedSet.add(sr);
942         }
943         return sortedSet.iterator();
944     }
