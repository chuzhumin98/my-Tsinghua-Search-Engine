160     /*** Resolves the filename for a settings object into a file path.
161      *
162      * It will also create the directory structure leading to this file
163      * if it doesn't exist.
164      *
165      * @param settings the settings object to get file path for.
166      * @return the file path for this settings object.
167      */
168     protected final File settingsToFilename(CrawlerSettings settings) {
169         File file;
170 
171         if (settings.getScope() == null || settings.getScope().equals("")) {
172             if (settings.isRefinement()) {
173                 file = new File(getSettingsDirectory(), File.separatorChar
174                         + REFINEMENT_DIR + File.separatorChar
175                         + settings.getName() + '.' + settingsFilenameSuffix);
176             } else {
177                 file = orderFile;
178             }
179         } else {
180             String elements[] = settings.getScope().split("//.");
181             if (elements.length == 0) {
182                 return orderFile;
183             }
184 
185             StringBuffer path = new StringBuffer();
186             for (int i = elements.length - 1; i > 0; i--) {
187                 path.append(elements[i]);
188                 path.append(File.separatorChar);
189             }
190             path.append(elements[0]);
191 
192             if (settings.isRefinement()) {
193                 file = new File(getSettingsDirectory(), path.toString()
194                         + File.separatorChar + REFINEMENT_DIR
195                         + File.separatorChar + settings.getName() + '.'
196                         + settingsFilenameSuffix);
197             } else {
198                 file = new File(getSettingsDirectory(), path.toString()
199                         + File.separatorChar + settingsFilename + "."
200                         + settingsFilenameSuffix);
201             }
202         }
203         return file;
204     }

211     /*** Write a CrawlerSettings object to a specified file.
212      *
213      * This method is similar to {@link #writeSettingsObject(CrawlerSettings)}
214      * except that it uses the submitted File object instead of trying to
215      * resolve where the file should be written.
216      *
217      * @param settings the settings object to be serialized.
218      * @param filename the file to which the settings object should be written.
219      */
220     public final void writeSettingsObject(
221             CrawlerSettings settings, File filename) {
222 
223         logger.fine("Writing " + filename.getAbsolutePath());
224         filename.getParentFile().mkdirs();
225 
226         FileOutputStream fos = null;
227         try {
228             long lastSaved = 0L;
229             File backup = null;
230             if (getOrder().getController() != null && filename.exists()) {
231                 // The crawler is running and file exists - make backup first.
232                 String name = filename.getName();
233                 lastSaved = settings.getLastSavedTime().getTime();
234                 name = name.substring(0, name.lastIndexOf('.')) + '_'
235                         + ArchiveUtils.get14DigitDate(lastSaved) + "."
236                         + settingsFilenameSuffix;
237                 backup = new File(filename.getParentFile(), name);
238                 FileUtils.copyFiles(filename, backup);
239             }
240 
241             fos = new FileOutputStream(filename);
242             StreamResult result =
243                 new StreamResult(
244                     new BufferedOutputStream(fos));
245             Transformer transformer =
246                 TransformerFactory.newInstance().newTransformer();
247             Source source = new CrawlSettingsSAXSource(settings);
248             transformer.transform(source, result);
249 
250             // Hack to get rid of unnesessary backupfiles.
251             // What happens is that the WUI often saves settings files
252             // several times during a settings change. This code removes the
253             // last backup file if its no more than 2 minutes old.
254             if (lastSaved > (System.currentTimeMillis() - 2 * 60 * 1000)) {
255                 backup.delete();
256             }
257         } catch (Exception e) {
258             e.printStackTrace();
259         } finally {
260             IOUtils.closeQuietly(fos);
261         }
262     }

264     /*** Read the CrawlerSettings object from a specific file.
265      *
266      * @param settings the settings object to be updated with data from the
267      *                 persistent storage.
268      * @param f the file to read from.
269      * @return the updated settings object or null if there was no data for this
270      *         in the persistent storage.
271      */    
272     protected final CrawlerSettings readSettingsObject(CrawlerSettings settings,
273             File f) {
274         CrawlerSettings result = null;
275         try {
276             InputStream is = null;
277             if (!f.exists()) {
278                 // Perhaps the file we're looking for is on the CLASSPATH.
279                 // DON'T look on the CLASSPATH for 'settings.xml' files.  The
280                 // look for 'settings.xml' files happens frequently. Not looking
281                 // on classpath for 'settings.xml' is an optimization based on
282                 // ASSUMPTION that there will never be a 'settings.xml' saved
283                 // on classpath.
284                 if (!f.getName().startsWith(settingsFilename)) {
285                     is = XMLSettingsHandler.class.
286                         getResourceAsStream(toResourcePath(f));
287                 }
288             } else {
289                 is = new FileInputStream(f);
290             }
291             if (is != null) {
292                 XMLReader parser = SAXParserFactory.newInstance()
293                     .newSAXParser().getXMLReader();
294                 InputStream file = new BufferedInputStream(is);
295                 parser.setContentHandler(new CrawlSettingsSAXHandler(settings));
296                 InputSource source = new InputSource(file);
297                 source.setSystemId(f.toURL().toExternalForm());
298                 parser.parse(source);
299                 result = settings;
300             }
301         } catch (SAXParseException e) {
302             logger.warning(e.getMessage() + " in '" + e.getSystemId()
303                 + "', line: " + e.getLineNumber() + ", column: "
304                 + e.getColumnNumber());
305         } catch (SAXException e) {
306             logger.warning(e.getMessage() + ": "
307                 + e.getException().getMessage());
308         } catch (ParserConfigurationException e) {
309             logger.warning(e.getMessage() + ": "
310                 + e.getCause().getMessage());
311         } catch (FactoryConfigurationError e) {
312             logger.warning(e.getMessage() + ": "
313                 + e.getException().getMessage());
314         } catch (IOException e) {
315             logger.warning("Could not access file '"
316                 + f.getAbsolutePath() + "': " + e.getMessage());
317         }
318         return result;
319     }

322      * Convert a File to a path that might be resolved from classpath/JAR
323      * resource sources. Such paths use linux-like path-separators. 
324      * 
325      * @param f File 
326      * @return path, shorn of any Windows-specific drive identifiers
327      */
328     public static String toResourcePath(File f) {
329         String path = f.toURI().getPath(); 
330         if(path.matches("^/[A-Z]:/.*")) {
331             // remove Windows drive-prefix, if any
332             path = path.substring(3); 
333         }
334         return path;

350     /*** Creates a replica of the settings file structure in another directory
351      * (fully recursive, includes all per host settings). The SettingsHandler
352      * will then refer to the new files.
353      *
354      * Observe that this method should only be called after the SettingsHandler
355      * has been initialized.
356      *
357      * @param newOrderFileName where the new order file should be saved.
358      * @param newSettingsDirectory the top level directory of the per host/domain
359      *                          settings files.
360      * @throws IOException
361      */
362     public void copySettings(File newOrderFileName, String newSettingsDirectory)
363       throws IOException {
364         File oldSettingsDirectory = getSettingsDirectory();
365 
366         // Write new orderfile and point the settingshandler to it
367         orderFile = newOrderFileName;
368         try {
369             getOrder().setAttribute(
370                 new Attribute(
371                     CrawlOrder.ATTR_SETTINGS_DIRECTORY, newSettingsDirectory));
372         } catch (Exception e) {
373             throw new IOException("Could not update settings with new location: "
374                 + e.getMessage());
375         }
376         writeSettingsObject(getSettingsObject(null));
377 
378         File newDir = getPathRelativeToWorkingDirectory(newSettingsDirectory);
379 
380         // Copy the per host files if src and dest directories are different.
381         if (oldSettingsDirectory.compareTo(newDir) != 0) {
382             FileUtils.copyFiles(oldSettingsDirectory, newDir);
383         }
384     }

405     public Collection getDomainOverrides(String rootDomain) {
406         File settingsDir = getSettingsDirectory();
407 
408         //Find the right start directory.
409         ArrayList<String> domains = new ArrayList<String>();
410         //First we deconstruct the rootDomain string
411         while(rootDomain != null && rootDomain.length()>0){
412             if(rootDomain.indexOf('.')<0){
413                 // Last level.
414                 domains.add(rootDomain);
415                 break; //We're done.
416             } else {
417                 // Got more then one level left.
418                 domains.add(rootDomain.substring(0,rootDomain.indexOf('.')));
419                 // Strip down rootDomain.
420                 rootDomain = rootDomain.substring(rootDomain.indexOf('.')+1);
421             }
422         }
423         //Build up a proper path
424         //Since the domains are right to left, we start at the end of the array.
425         StringBuffer subDir = new StringBuffer();
426         for(int i=(domains.size()-1) ; i>=0 ; i--){
427             subDir.append(File.separator+domains.get(i));
428         }
429         //Then we move to the approprite directory.
430         settingsDir = new File(settingsDir.getPath()+subDir);
431         TreeSet<String> confirmedSubDomains = new TreeSet<String>();
432         if(settingsDir.exists()){
433             // Found our place! Search through it's subdirs.
434             File[] possibleSubDomains = settingsDir.listFiles();
435             for (int i = 0; i < possibleSubDomains.length; i++) {
436                 if (possibleSubDomains[i].isDirectory()
437                     && isOverride(possibleSubDomains[i])) {
438                     // Found one!
439                     confirmedSubDomains.add(possibleSubDomains[i].getName());
440                 }
441             }
442         }
443         return confirmedSubDomains;
444     }

446     /***
447      * Checks if a file is a a 'per host' override or if it's a directory if it
448      * or it's subdirectories  contains a 'per host' override file.
449      * @param f The file or directory to check
450      * @return True if the file is an override or it's a directory that contains
451      *         such a file.
452      */
453     private boolean isOverride(File f){
454         if(f.isDirectory()){
455             // Have a directory, check it's contents.
456             File[] subs = f.listFiles();
457             for(int i=0 ; i < subs.length ; i++){
458                 if(isOverride(subs[i])){
459                     // Found one. Can stop looking.
460                     return true;
461                 }
462             }
463         } else if (f.getName().equals(
464                 settingsFilename + "." + settingsFilenameSuffix)) {
465             // This is an override file (or sure looks like one in any case).
466             return true;
467         }
468         // Didn't find an override.
469         return false;
470     }

472     /*** Delete a settings object from persistent storage.
473      *
474      * Deletes the file represented by the submitted settings object. All empty
475      * directories that are parents to the files path are also deleted.
476      *
477      * @param settings the settings object to delete.
478      */
479     public void deleteSettingsObject(CrawlerSettings settings) {
480         super.deleteSettingsObject(settings);
481         File settingsDirectory = getSettingsDirectory();
482         File settingsFile = settingsToFilename(settings);
483 
484         if(!settingsFile.delete()) {
485             throw new RuntimeException("Could not delete: "+settingsFile);
486         }
487         settingsFile = settingsFile.getParentFile();
488         while (settingsFile.isDirectory() && settingsFile.list().length == 0
489                 && !settingsFile.equals(settingsDirectory)) {
490             if(!settingsFile.delete()) {
491                 logger.warning("Could not delete: "+settingsFile);
492             }
493             settingsFile = settingsFile.getParentFile();
494         }
495     }

500     public List<String> getListOfAllFiles() {
501         ArrayList<String> list = new ArrayList<String>();
502         // Add CrawlOrder.
503         list.add(getOrderFile().getAbsolutePath());
504         // Iterate through the entire override hierarchy
505         if (getSettingsDirectory().exists()) {
506             recursiveFindFiles(getSettingsDirectory(),list);
507         }
508         // Get files used by settings modules.
509         recursiveFindSecondaryFiles(getOrder(),list);
510         return list;

513     /***
514      * Add any files being used by any of the Modules making up the settings to
515      * the list.
516      *
517      * @param mbean A ModuleType to interrogate for files. Any child modules
518      *           will be recursively interrogated.
519      * @param list The list to add found files to.
520      */
521     private void recursiveFindSecondaryFiles(ComplexType mbean, 
522             ArrayList<String> list) {
523         MBeanInfo info = mbean.getMBeanInfo();
524         MBeanAttributeInfo[] a = info.getAttributes();
525         // Interrogate the current module
526         if(mbean instanceof ModuleType){
527             ((ModuleType)mbean).listUsedFiles(list);
528         }
529 
530         // Recursively interrogate all sub modules that are of ModuleType
531         for(int n=0; n<a.length; n++) {
532             if(a[n] == null) {
533                 // Error null attribute.
534             } else {
535                 ModuleAttributeInfo att = (ModuleAttributeInfo)a[n];
536                 Object currentAttribute;
537                 try {
538                     currentAttribute = mbean.getAttribute(att.getName());
539                     if(currentAttribute instanceof ComplexType) {
540                         recursiveFindSecondaryFiles((ComplexType)currentAttribute,list);
541                     }
542                 } catch (AttributeNotFoundException e) {
543                     // TODO Auto-generated catch block
544                     e.printStackTrace();
545                 } catch (MBeanException e) {
546                     // TODO Auto-generated catch block
547                     e.printStackTrace();
548                 } catch (ReflectionException e) {
549                     // TODO Auto-generated catch block
550                     e.printStackTrace();
551                 }
552             }
553         }
554     }

556     /***
557      * Starting at the specific directory this method will iterate through all
558      * sub directories and add each file (as absolute name, with path as a
559      * string) to the provided ArrayList. Any file found under the settings
560      * directory with the proper suffix will be considered valid and added to
561      * the list.
562      * @param dir Starting directory
563      * @param list The list to add to
564      */
565     private void recursiveFindFiles(File dir, ArrayList<String> list){
566         File[] subs = dir.listFiles();
567         if (subs != null) {
568             for(int i=0 ; i < subs.length ; i++){
569                 if(subs[i].isDirectory()){
570                     recursiveFindFiles(subs[i],list);
571                 } else {
572                     if(subs[i].getName().endsWith(settingsFilenameSuffix)){
573                         // Add it to list
574                         list.add(subs[i].getAbsolutePath());
575                     }
576                 }
577             }
578         }
