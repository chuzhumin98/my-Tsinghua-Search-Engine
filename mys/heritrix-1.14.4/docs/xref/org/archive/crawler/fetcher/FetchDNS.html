107     protected void innerProcess(CrawlURI curi) {
108         if (!curi.getUURI().getScheme().equals("dns")) {
109             // Only handles dns
110             return;
111         }
112         Record[] rrecordSet = null; // Retrieved dns records
113         String dnsName = null;
114         try {
115             dnsName = curi.getUURI().getReferencedHost();
116         } catch (URIException e) {
117             logger.log(Level.SEVERE, "Failed parse of dns record " + curi, e);
118         }
119         
120         if(dnsName == null) {
121             curi.setFetchStatus(S_UNFETCHABLE_URI);
122             return;
123         }
124 
125         // Make sure we're in "normal operating mode", e.g. a cache +
126         // controller exist to assist us.
127         CrawlHost targetHost = null;
128         if (getController() != null &&
129                 getController().getServerCache() != null) {
130             targetHost = getController().getServerCache().getHostFor(dnsName);
131         } else {
132             // Standalone operation (mostly for test cases/potential other uses)
133             targetHost = new CrawlHost(dnsName);
134         }
135         if (isQuadAddress(curi, dnsName, targetHost)) {
136         	// We're done processing.
137         	return;
138         }
139         
140         // Do actual DNS lookup.
141         curi.putLong(A_FETCH_BEGAN_TIME, System.currentTimeMillis());
142 
143         // Try to get the records for this host (assume domain name)
144         // TODO: Bug #935119 concerns potential hang here
145         try {
146             rrecordSet = (new Lookup(dnsName, TypeType, ClassType)).run();
147         } catch (TextParseException e) {
148             rrecordSet = null;
149         }
150         curi.setContentType("text/dns");
151         if (rrecordSet != null) {
152             if (logger.isLoggable(Level.FINE)) {
153                 logger.fine("Found recordset for " + dnsName);
154             }
155         	storeDNSRecord(curi, dnsName, targetHost, rrecordSet);
156         } else {
157             if (logger.isLoggable(Level.FINE)) {
158                 logger.fine("Failed find of recordset for " + dnsName);
159             }
160             if (((Boolean)getUncheckedAttribute(null,
161                     ATTR_ACCEPT_NON_DNS_RESOLVES)).booleanValue()) {
162                 // Do lookup that bypasses javadns.
163                 InetAddress address = null;
164                 try {
165                     address = InetAddress.getByName(dnsName);
166                 } catch (UnknownHostException e1) {
167                     address = null;
168                 }
169                 if (address != null) {
170                     targetHost.setIP(address, DEFAULT_TTL_FOR_NON_DNS_RESOLVES);
171                     curi.setFetchStatus(S_GETBYNAME_SUCCESS);
172                     if (logger.isLoggable(Level.FINE)) {
173                         logger.fine("Found address for " + dnsName +
174                             " using native dns.");
175                     }
176                 } else {
177                     if (logger.isLoggable(Level.FINE)) {
178                         logger.fine("Failed find of address for " + dnsName +
179                             " using native dns.");
180                     }
181                     setUnresolvable(curi, targetHost);
182                 }
183             } else {
184                 setUnresolvable(curi, targetHost);
185             }
186         }
187         curi.putLong(A_FETCH_COMPLETED_TIME, System.currentTimeMillis());
188     }

190     protected void storeDNSRecord(final CrawlURI curi, final String dnsName,
191     		final CrawlHost targetHost, final Record[] rrecordSet) {
192         // Get TTL and IP info from the first A record (there may be
193         // multiple, e.g. www.washington.edu) then update the CrawlServer
194         ARecord arecord = getFirstARecord(rrecordSet);
195         if (arecord == null) {
196             throw new NullPointerException("Got null arecord for " +
197                 dnsName);
198         }
199         targetHost.setIP(arecord.getAddress(), arecord.getTTL());
200         try {
201         	recordDNS(curi, rrecordSet);
202             curi.setFetchStatus(S_DNS_SUCCESS);
203             curi.putString(A_DNS_SERVER_IP_LABEL, ResolverConfig.getCurrentConfig().server());
204         } catch (IOException e) {
205         	logger.log(Level.SEVERE, "Failed store of DNS Record for " +
206         		curi.toString(), e);
207         	setUnresolvable(curi, targetHost);
208         }
209     }

211     protected boolean isQuadAddress(final CrawlURI curi, final String dnsName,
212 			final CrawlHost targetHost) {
213 		boolean result = false;
214 		Matcher matcher = InetAddressUtil.IPV4_QUADS.matcher(dnsName);
215 		// If it's an ip no need to do a lookup
216 		if (matcher == null || !matcher.matches()) {
217 			return result;
218 		}
219 		
220 		result = true;
221 		// Ideally this branch would never be reached: no CrawlURI
222 		// would be created for numerical IPs
223 		if (logger.isLoggable(Level.WARNING)) {
224 			logger.warning("Unnecessary DNS CrawlURI created: " + curi);
225 		}
226 		try {
227 			targetHost.setIP(InetAddress.getByAddress(dnsName, new byte[] {
228 					(byte) (new Integer(matcher.group(1)).intValue()),
229 					(byte) (new Integer(matcher.group(2)).intValue()),
230 					(byte) (new Integer(matcher.group(3)).intValue()),
231 					(byte) (new Integer(matcher.group(4)).intValue()) }),
232 					CrawlHost.IP_NEVER_EXPIRES); // Never expire numeric IPs
233 			curi.setFetchStatus(S_DNS_SUCCESS);
234 		} catch (UnknownHostException e) {
235 			logger.log(Level.SEVERE, "Should never be " + e.getMessage(), e);
236 			setUnresolvable(curi, targetHost);
237 		}
238 		return result;
239 	}

241     protected void recordDNS(final CrawlURI curi, final Record[] rrecordSet)
242 	throws IOException {
243 		final byte[] dnsRecord =
244 			getDNSRecord(curi.getLong(A_FETCH_BEGAN_TIME), rrecordSet);
245 		HttpRecorder rec = HttpRecorder.getHttpRecorder();
246         
247         // Shall we get a digest on the content downloaded?
248 		boolean digestContent  = ((Boolean)getUncheckedAttribute(curi,
249                 FetchHTTP.ATTR_DIGEST_CONTENT)).booleanValue();
250         String algorithm = null; 
251         if (digestContent) {
252             algorithm = ((String)getUncheckedAttribute(curi,
253                 FetchHTTP.ATTR_DIGEST_ALGORITHM));
254             rec.getRecordedInput().setDigest(algorithm);
255         } else {
256             // clear
257             rec.getRecordedInput().setDigest((MessageDigest)null);
258         }
259         
260 		curi.setHttpRecorder(rec);
261 		InputStream is = curi.getHttpRecorder().inputWrap(
262 				new ByteArrayInputStream(dnsRecord));
263         if(digestContent) {
264             rec.getRecordedInput().startDigest();
265         }
266 		// Reading from the wrapped stream, behind the scenes, will write
267 		// files into scratch space
268 		try {
269 			while (is.read(this.reusableBuffer) != -1) {
270 				continue;
271 			}
272 		} finally {
273 			is.close();
274 			rec.closeRecorders();
275 		}
276 		curi.setContentSize(dnsRecord.length);
277         if (digestContent) {
278             curi.setContentDigest(algorithm,
279                 rec.getRecordedInput().getDigestValue());
280         }
281 	}

283     protected byte [] getDNSRecord(final long fetchStart,
284     		final Record[] rrecordSet)
285     throws IOException {
286         ByteArrayOutputStream baos = new ByteArrayOutputStream();
287         // Start the record with a 14-digit date per RFC 2540
288         byte[] fetchDate = ArchiveUtils.get14DigitDate(fetchStart).getBytes();
289         baos.write(fetchDate);
290         // Don't forget the newline
291         baos.write("\n".getBytes());
292         int recordLength = fetchDate.length + 1;
293         if (rrecordSet != null) {
294             for (int i = 0; i < rrecordSet.length; i++) {
295                 byte[] record = rrecordSet[i].toString().getBytes();
296                 recordLength += record.length;
297                 baos.write(record);
298                 // Add the newline between records back in
299                 baos.write("\n".getBytes());
300                 recordLength += 1;
301             }
302         }
303         return baos.toByteArray();
304     }

306     protected void setUnresolvable(CrawlURI curi, CrawlHost host) {
307         host.setIP(null, 0);
308         curi.setFetchStatus(S_DOMAIN_UNRESOLVABLE); 
309     }

311     protected ARecord getFirstARecord(Record[] rrecordSet) {
312         ARecord arecord = null;
313         if (rrecordSet == null || rrecordSet.length == 0) {
314             if (logger.isLoggable(Level.FINEST)) {
315                 logger.finest("rrecordSet is null or zero length: " +
316                     rrecordSet);
317             }
318             return arecord;
319         }
320         for (int i = 0; i < rrecordSet.length; i++) {
321             if (rrecordSet[i].getType() != Type.A) {
322                 if (logger.isLoggable(Level.FINEST)) {
323                     logger.finest("Record " + Integer.toString(i) +
324                         " is not A type but " + rrecordSet[i].getType());
325                 }
326                 continue;
327             }
328             arecord = (ARecord) rrecordSet[i];
329             break;
330         }
331         return arecord;
332     }
