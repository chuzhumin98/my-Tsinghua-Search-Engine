225     /***
226      * Processes the given URI.  If the given URI is not an FTP URI, then
227      * this method does nothing.  Otherwise an attempt is made to connect
228      * to the FTP server.
229      * 
230      * <p>If the connection is successful, an attempt will be made to CD to 
231      * the path specified in the URI.  If the remote CD command succeeds, 
232      * then it is assumed that the URI represents a directory.  If the
233      * CD command fails, then it is assumed that the URI represents
234      * a file.
235      * 
236      * <p>For directories, the directory listing will be fetched using
237      * the FTP LIST command, and saved to the HttpRecorder.  If the
238      * <code>extract.from.dirs</code> attribute is set to true, then
239      * the files in the fetched list will be added to the curi as
240      * extracted FTP links.  (It was easier to do that here, rather
241      * than writing a separate FTPExtractor.)
242      * 
243      * <p>For files, the file will be fetched using the FTP RETR
244      * command, and saved to the HttpRecorder.
245      * 
246      * <p>All file transfers (including directory listings) occur using
247      * Binary mode transfer.  Also, the local passive transfer mode
248      * is always used, to play well with firewalls.
249      * 
250      * @param curi  the curi to process
251      * @throws InterruptedException  if the thread is interrupted during
252      *   processing
253      */
254     public void innerProcess(CrawlURI curi) throws InterruptedException {
255         if (!curi.getUURI().getScheme().equals("ftp")) {
256             return;
257         }
258         
259         curi.putLong(A_FETCH_BEGAN_TIME, System.currentTimeMillis());
260         ClientFTP client = new ClientFTP();
261         HttpRecorder recorder = HttpRecorder.getHttpRecorder();
262         
263         try {
264             if (logger.isLoggable(Level.INFO)) {
265                 logger.info("attempting to fetch ftp uri: " + curi);
266             }
267             fetch(curi, client, recorder);
268         } catch (IOException e) {
269             if (logger.isLoggable(Level.INFO)) {
270                 logger.info(curi + ": " + e);
271             }
272             curi.addLocalizedError(this.getName(), e, "uhhhh");
273             curi.setFetchStatus(S_CONNECT_FAILED);
274         } finally {
275             disconnect(client);
276             curi.putLong(A_FETCH_COMPLETED_TIME, System.currentTimeMillis());
277             curi.putString(A_FTP_CONTROL_CONVERSATION, client.getControlConversation());
278             if (logger.isLoggable(Level.INFO)) {
279                 logger.info("finished attempt to fetch ftp uri: " + curi);
280             }
281         }
282     }

285     /***
286      * Fetches a document from an FTP server.
287      * 
288      * @param curi      the URI of the document to fetch
289      * @param client    the FTPClient to use for the fetch
290      * @param recorder  the recorder to preserve the document in
291      * @throws IOException  if a network or protocol error occurs
292      * @throws InterruptedException  if the thread is interrupted
293      */
294     private void fetch(CrawlURI curi, ClientFTP client, HttpRecorder recorder) 
295     throws IOException, InterruptedException {
296         // Connect to the FTP server.
297         UURI uuri = curi.getUURI();
298         int port = uuri.getPort();
299         if (port == -1) {
300             port = 21;
301         }
302         
303         client.connect(uuri.getHost(), port);
304         
305         // Authenticate.
306         String[] auth = getAuth(curi);
307         client.login(auth[0], auth[1]);
308         
309         // The given resource may or may not be a directory.
310         // To figure out which is which, execute a CD command to
311         // the UURI's path.  If CD works, it's a directory.
312         boolean isDirectory = client.changeWorkingDirectory(uuri.getPath());
313         
314         // Get a data socket.  This will either be the result of a NLST
315         // command for a directory, or a RETR command for a file.
316         int command;
317         String path;
318         if (isDirectory) {
319             curi.addAnnotation("ftpDirectoryList");
320             command = FTPCommand.NLST;
321             client.setFileType(FTP.ASCII_FILE_TYPE);
322             path = ".";
323         } else { 
324             command = FTPCommand.RETR;
325             client.setFileType(FTP.BINARY_FILE_TYPE);
326             path = uuri.getPath();
327         }
328 
329         client.enterLocalPassiveMode();
330         Socket socket = null;
331 
332         try {
333             socket = client.openDataConnection(command, path);
334 
335             // if "227 Entering Passive Mode" these will get reset later
336             curi.setFetchStatus(client.getReplyCode());
337             curi.putString(A_FTP_FETCH_STATUS, client.getReplyStrings()[0]);
338 
339         } catch (IOException e) {
340             // try it again, see AbstractFrontier.needsRetrying()
341             curi.setFetchStatus(S_CONNECT_LOST);
342         }
343 
344         // Save the streams in the CURI, where downstream processors
345         // expect to find them.
346         if (socket != null) {
347             // Shall we get a digest on the content downloaded?
348             boolean digestContent  = ((Boolean)getUncheckedAttribute(curi,
349                     FetchHTTP.ATTR_DIGEST_CONTENT)).booleanValue();
350             String algorithm = null; 
351             if (digestContent) {
352                 algorithm = ((String)getUncheckedAttribute(curi,
353                     FetchHTTP.ATTR_DIGEST_ALGORITHM));
354                 recorder.getRecordedInput().setDigest(algorithm);
355                 recorder.getRecordedInput().startDigest();
356             } else {
357                 // clear
358                 recorder.getRecordedInput().setDigest((MessageDigest)null);
359             }
360                     
361             try {
362                 saveToRecorder(curi, socket, recorder);
363             } finally {
364                 recorder.close();
365                 client.closeDataConnection(); // does socket.close()
366                 curi.setContentSize(recorder.getRecordedInput().getSize());
367 
368                 // "226 Transfer complete."
369                 client.getReply();
370                 curi.setFetchStatus(client.getReplyCode());
371                 curi.putString(A_FTP_FETCH_STATUS, client.getReplyStrings()[0]);
372                 
373                 if (isDirectory) {
374                     curi.setContentType("text/plain");
375                 } else {
376                     curi.setContentType("application/octet-stream");
377                 }
378                 
379                 if (logger.isLoggable(Level.INFO)) {
380                     logger.info("read " + recorder.getRecordedInput().getSize()
381                             + " bytes from ftp data socket");
382                 }
383 
384                 if (digestContent) {
385                     curi.setContentDigest(algorithm,
386                         recorder.getRecordedInput().getDigestValue());
387                 }
388             }
389 
390             if (isDirectory) {
391                 extract(curi, recorder);
392             }
393         }
394 
395         addParent(curi);
396     }

398     /***
399      * Saves the given socket to the given recorder.
400      * 
401      * @param curi      the curi that owns the recorder
402      * @param socket    the socket whose streams to save
403      * @param recorder  the recorder to save them to
404      * @throws IOException  if a network or file error occurs
405      * @throws InterruptedException  if the thread is interrupted
406      */
407     private void saveToRecorder(CrawlURI curi,
408             Socket socket, HttpRecorder recorder) 
409     throws IOException, InterruptedException {
410         curi.setHttpRecorder(recorder);
411         recorder.inputWrap(socket.getInputStream());
412         recorder.outputWrap(socket.getOutputStream());
413         recorder.markContentBegin();
414 
415         // Read the remote file/dir listing in its entirety.
416         long softMax = 0;
417         long hardMax = getMaxLength(curi);
418         long timeout = (long)getTimeout(curi) * 1000;
419         int maxRate = getFetchBandwidth(curi);
420         RecordingInputStream input = recorder.getRecordedInput();
421         input.setLimits(hardMax, timeout, maxRate); 
422         input.readFullyOrUntil(softMax);
423     }

426     /***
427      * Extract FTP links in a directory listing.
428      * The listing must already be saved to the given recorder.
429      * 
430      * @param curi      The curi to save extracted links to
431      * @param recorder  The recorder containing the directory listing
432      */
433     private void extract(CrawlURI curi, HttpRecorder recorder) {
434         if (!getExtractFromDirs(curi)) {
435             return;
436         }
437         
438         ReplayCharSequence seq = null;
439         try {
440             seq = recorder.getReplayCharSequence();
441             extract(curi, seq);
442         } catch (IOException e) {
443             logger.log(Level.SEVERE, "IO error during extraction.", e);
444         } catch (RuntimeException e) {
445             logger.log(Level.SEVERE, "IO error during extraction.", e);
446         } finally {
447             close(seq);
448         }
449     }

452     /***
453      * Extracts FTP links in a directory listing.
454      * 
455      * @param curi  The curi to save extracted links to
456      * @param dir   The directory listing to extract links from
457      * @throws URIException  if an extracted link is invalid
458      */
459     private void extract(CrawlURI curi, ReplayCharSequence dir) {
460         logger.log(Level.FINEST, "Extracting URIs from FTP directory.");
461         Matcher matcher = DIR.matcher(dir);
462         while (matcher.find()) {
463             String file = matcher.group(1);
464             addExtracted(curi, file);
465         }
466     }

469     /***
470      * Adds an extracted filename to the curi.  A new URI will be formed
471      * by taking the given curi (which should represent the directory the
472      * file lives in) and appending the file.
473      * 
474      * @param curi  the curi to store the discovered link in
475      * @param file  the filename of the discovered link
476      */
477     private void addExtracted(CrawlURI curi, String file) {
478         try {
479             file = URLEncoder.encode(file, "UTF-8");
480         } catch (UnsupportedEncodingException e) {
481             throw new AssertionError(e);
482         }
483         if (logger.isLoggable(Level.FINEST)) {
484             logger.log(Level.FINEST, "Found " + file);
485         }
486         String base = curi.toString();
487         if (base.endsWith("/")) {
488             base = base.substring(0, base.length() - 1);
489         }
490         try {
491             UURI n = UURIFactory.getInstance(base + "/" + file);
492             Link link = new Link(curi.getUURI(), n, NAVLINK_MISC, NAVLINK_HOP);
493             curi.addOutLink(link);
494         } catch (URIException e) {
495             logger.log(Level.WARNING, "URI error during extraction.", e);            
496         }
497     }

500     /***
501      * Extracts the parent URI from the given curi, then adds that parent
502      * URI as a discovered link to the curi. 
503      * 
504      * <p>If the <code>extract-parent</code> attribute is false, then this
505      * method does nothing.  Also, if the path of the given curi is 
506      * <code>/</code>, then this method does nothing.
507      * 
508      * <p>Otherwise the parent is determined by eliminated the lowest part
509      * of the URI's path.  Eg, the parent of <code>ftp://foo.com/one/two</code>
510      * is <code>ftp://foo.com/one</code>.
511      * 
512      * @param curi  the curi whose parent to add
513      */
514     private void addParent(CrawlURI curi) {
515         if (!getExtractParent(curi)) {
516             return;
517         }
518         UURI uuri = curi.getUURI();
519         try {
520             if (uuri.getPath().equals("/")) {
521                 // There's no parent to add.
522                 return;
523             }
524             String scheme = uuri.getScheme();
525             String auth = uuri.getEscapedAuthority();
526             String path = uuri.getEscapedCurrentHierPath();
527             UURI parent = UURIFactory.getInstance(scheme + "://" + auth + path);
528 
529             Link link = new Link(uuri, parent, NAVLINK_MISC, NAVLINK_HOP);
530             curi.addOutLink(link);
531         } catch (URIException e) {
532             logger.log(Level.WARNING, "URI error during extraction.", e);
533         }
534     }

596     /***
597      * Returns the username and password for the given URI.  This method
598      * always returns an array of length 2.  The first element in the returned
599      * array is the username for the URI, and the second element is the
600      * password.
601      * 
602      * <p>If the URI itself contains the username and password (i.e., it looks
603      * like <code>ftp://username:password@host/path</code>) then that username
604      * and password are returned.
605      * 
606      * <p>Otherwise the settings system is probed for the <code>username</code>
607      * and <code>password</code> attributes for this <code>FTPFetch</code>
608      * and the given <code>curi</code> context.  The values of those 
609      * attributes are then returned. If the FetchFTP password attribute is 
610      * not set, the crawl operator 'From' attribute is used.
611      * 
612      * @param curi  the curi whose username and password to return
613      * @return  an array containing the username and password
614      */
615     private String[] getAuth(CrawlURI curi) {
616         String[] result = new String[2];
617         UURI uuri = curi.getUURI();
618         String userinfo;
619         try {
620             userinfo = uuri.getUserinfo();
621         } catch (URIException e) {
622             assert false;
623             logger.finest("getUserinfo raised URIException.");
624             userinfo = null;
625         }
626         if (userinfo != null) {
627             int p = userinfo.indexOf(':');
628             if (p > 0) {
629                 result[0] = userinfo.substring(0,p);
630                 result[1] = userinfo.substring(p + 1);
631                 return result;
632             }
633         }
