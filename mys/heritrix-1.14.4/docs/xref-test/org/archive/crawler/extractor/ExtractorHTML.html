65       * <p>
66       * This pattern extracts either:
67       * <li> (1) whole &lt;script&gt;...&lt;/script&gt; or
68       * <li> (2) &lt;style&gt;...&lt;/style&gt; or
69       * <li> (3) &lt;meta ...&gt; or
70       * <li> (4) any other open-tag with at least one attribute
71       * (eg matches "&lt;a href='boo'&gt;" but not "&lt;/a&gt;" or "&lt;br&gt;")
72       * <p>
73       * groups:
74       * <li> 1: SCRIPT SRC=foo&gt;boo&lt;/SCRIPT
75       * <li> 2: just script open tag
76       * <li> 3: STYLE TYPE=moo&gt;zoo&lt;/STYLE
77       * <li> 4: just style open tag
78       * <li> 5: entire other tag, without '<' '>'
79       * <li> 6: element
80       * <li> 7: META
81       * <li> 8: !-- comment --
82       */

119     // this pattern extracts attributes from any open-tag innards
120     // matched by the above. attributes known to be URIs of various
121     // sorts are matched specially
122     static final String EACH_ATTRIBUTE_EXTRACTOR =
123       "(?is)//b((href)|(action)|(on//w*)" // 1, 2, 3, 4 
124      +"|((?:src)|(?:lowsrc)|(?:background)|(?:cite)|(?:longdesc)" // ...
125      +"|(?:usemap)|(?:profile)|(?:datasrc))" // 5
126      +"|(codebase)|((?:classid)|(?:data))|(archive)|(code)" // 6, 7, 8, 9
127      +"|(value)|(style)|(method)" // 10, 11, 12
128      +"|([-//w]{1,"+MAX_ATTR_NAME_LENGTH+"}))" // 13
129      +"//s*=//s*"
130      +"(?:(?:\"(.{0,"+MAX_ATTR_VAL_LENGTH+"}?)(?:\"|$))" // 14
131      +"|(?:'(.{0,"+MAX_ATTR_VAL_LENGTH+"}?)(?:'|$))" // 15
132      +"|(//S{1,"+MAX_ATTR_VAL_LENGTH+"}))"; // 16
133     // groups:
134     // 1: attribute name
135     // 2: HREF - single URI relative to doc base, or occasionally javascript:
136     // 3: ACTION - single URI relative to doc base, or occasionally javascript:
137     // 4: ON[WHATEVER] - script handler
138     // 5: SRC,LOWSRC,BACKGROUND,CITE,LONGDESC,USEMAP,PROFILE, or DATASRC
139     //    single URI relative to doc base
140     // 6: CODEBASE - a single URI relative to doc base, affecting other
141     //    attributes
142     // 7: CLASSID, DATA - a single URI relative to CODEBASE (if supplied)
143     // 8: ARCHIVE - one or more space-delimited URIs relative to CODEBASE
144     //    (if supplied)
145     // 9: CODE - a single URI relative to the CODEBASE (is specified).
146     // 10: VALUE - often includes a uri path on forms
147     // 11: STYLE - inline attribute style info
148     // 12: METHOD - form GET/POST
149     // 13: any other attribute
150     // 14: double-quote delimited attr value
151     // 15: single-quote delimited attr value
152     // 16: space-delimited attr value

236     protected void processGeneralTag(CrawlURI curi, CharSequence element,
237             CharSequence cs) {
238 
239         Matcher attr = TextUtils.getMatcher(EACH_ATTRIBUTE_EXTRACTOR, cs);
240 
241         // Just in case it's an OBJECT or APPLET tag
242         String codebase = null;
243         ArrayList<String> resources = null;
244         
245         // Just in case it's a FORM
246         CharSequence action = null;
247         CharSequence actionContext = null;
248         CharSequence method = null; 
249         
250         // Just in case it's a VALUE whose interpretation depends on accompanying NAME
251         CharSequence valueVal = null; 
252         CharSequence valueContext = null;
253         CharSequence nameVal = null; 
254         
255         final boolean framesAsEmbeds = ((Boolean)getUncheckedAttribute(curi,
256             ATTR_TREAT_FRAMES_AS_EMBED_LINKS)).booleanValue();
257 
258         final boolean ignoreFormActions = ((Boolean)getUncheckedAttribute(curi,
259                 ATTR_IGNORE_FORM_ACTION_URLS)).booleanValue();
260         
261         final boolean extractValueAttributes = ((Boolean)getUncheckedAttribute
262                 (curi, EXTRACT_VALUE_ATTRIBUTES)).booleanValue();
263         
264         final String elementStr = element.toString();
265         while (attr.find()) {
266             int valueGroup =
267                 (attr.start(14) > -1) ? 14 : (attr.start(15) > -1) ? 15 : 16;
268             int start = attr.start(valueGroup);
269             int end = attr.end(valueGroup);
270             assert start >= 0: "Start is: " + start + ", " + curi;
271             assert end >= 0: "End is :" + end + ", " + curi;
272             CharSequence value = cs.subSequence(start, end);
273             CharSequence attrName = cs.subSequence(attr.start(1),attr.end(1));
274             value = TextUtils.unescapeHtml(value);
275             if (attr.start(2) > -1) {
276                 // HREF
277                 CharSequence context =
278                     Link.elementContext(element, attr.group(2));
279                 if(elementStr.equalsIgnoreCase(LINK)) {
280                     // <LINK> elements treated as embeds (css, ico, etc)
281                     processEmbed(curi, value, context);
282                 } else {
283                     // other HREFs treated as links
284                     processLink(curi, value, context);
285                 }
286                 if (elementStr.equalsIgnoreCase(BASE)) {
287                     try {
288                         curi.setBaseURI(value.toString());
289                     } catch (URIException e) {
290                         if (getController() != null) {
291                             // Controller can be null: e.g. when running
292                             // ExtractorTool.
293                             getController().logUriError(e, curi.getUURI(),
294                                 value.toString());
295                         } else {
296                             logger.info("Failed set base uri: " +
297                                 curi + ", " + value.toString() + ": " +
298                                 e.getMessage());
299                         }
300                     }
301                 }
302             } else if (attr.start(3) > -1) {
303                 // ACTION
304                 if (!ignoreFormActions) {
305                     action = value; 
306                     actionContext = Link.elementContext(element,
307                         attr.group(3));
308                     // handling finished only at end (after METHOD also collected)
309                 }
310             } else if (attr.start(4) > -1) {
311                 // ON____
312                 processScriptCode(curi, value); // TODO: context?
313             } else if (attr.start(5) > -1) {
314                 // SRC etc.
315                 CharSequence context = Link.elementContext(element,
316                     attr.group(5));
317                 
318                 // true, if we expect another HTML page instead of an image etc.
319                 // TODO: add explicit 'F'rame hop type? (it's not really L, and
320                 // different enough from other 'E's)
321                 final char hopType;
322                 
323                 if(!framesAsEmbeds
324                     && (elementStr.equalsIgnoreCase(FRAME) || elementStr
325                         .equalsIgnoreCase(IFRAME))) {
326                     hopType = Link.NAVLINK_HOP;
327                 } else {
328                     hopType = Link.EMBED_HOP;
329                 }
330                 processEmbed(curi, value, context, hopType);
331             } else if (attr.start(6) > -1) {
332                 // CODEBASE
333                 codebase = (value instanceof String)?
334                     (String)value: value.toString();
335                 CharSequence context = Link.elementContext(element,
336                     attr.group(6));
337                 processEmbed(curi, codebase, context);
338             } else if (attr.start(7) > -1) {
339                 // CLASSID, DATA
340                 if (resources == null) {
341                     resources = new ArrayList<String>();
342                 }
343                 resources.add(value.toString());
344             } else if (attr.start(8) > -1) {
345                 // ARCHIVE
346                 if (resources==null) {
347                     resources = new ArrayList<String>();
348                 }
349                 String[] multi = TextUtils.split(WHITESPACE, value);
350                 for(int i = 0; i < multi.length; i++ ) {
351                     resources.add(multi[i]);
352                 }
353             } else if (attr.start(9) > -1) {
354                 // CODE
355                 if (resources==null) {
356                     resources = new ArrayList<String>();
357                 }
358                 // If element is applet and code value does not end with
359                 // '.class' then append '.class' to the code value.
360                 if (elementStr.equalsIgnoreCase(APPLET) &&
361                         !value.toString().toLowerCase().endsWith(CLASSEXT)) {
362                     resources.add(value.toString() + CLASSEXT);
363                 } else {
364                     resources.add(value.toString());
365                 }
366             } else if (attr.start(10) > -1) {
367                 // VALUE, with possibility of URI
368                 // store value, context for handling at end
369                 valueVal = value; 
370                 valueContext = Link.elementContext(element,attr.group(10));
371             } else if (attr.start(11) > -1) {
372                 // STYLE inline attribute
373                 // then, parse for URIs
374                 this.numberOfLinksExtracted += ExtractorCSS.processStyleCode(
375                     curi, value, getController());
376                 
377             } else if (attr.start(12) > -1) {
378                 // METHOD
379                 method = value;
380                 // form processing finished at end (after ACTION also collected)
381             } else if (attr.start(13) > -1) {
382                 if("NAME".equalsIgnoreCase(attrName.toString())) {
383                     // remember 'name' for end-analysis
384                     nameVal = value; 
385                 }
386                 if("FLASHVARS".equalsIgnoreCase(attrName.toString())) {
387                     // consider FLASHVARS attribute immediately
388                     valueContext = Link.elementContext(element,attr.group(13));
389                     considerQueryStringValues(curi, value, valueContext,Link.SPECULATIVE_HOP);
390                 }
391                 // any other attribute
392                 // ignore for now
393                 // could probe for path- or script-looking strings, but
394                 // those should be vanishingly rare in other attributes,
395                 // and/or symptomatic of page bugs
396             }
397         }
398         TextUtils.recycleMatcher(attr);
399 
400         // finish handling codebase/resources now that all available
401         if (resources != null) {
402             Iterator<String> iter = resources.iterator();
403             UURI codebaseURI = null;
404             String res = null;
405             try {
406                 if (codebase != null) {
407                     // TODO: Pass in the charset.
408                     codebaseURI = UURIFactory.
409                         getInstance(curi.getUURI(), codebase);
410                 }
411                 while(iter.hasNext()) {
412                     res = iter.next().toString();
413                     res = (String) TextUtils.unescapeHtml(res);
414                     if (codebaseURI != null) {
415                         res = codebaseURI.resolve(res).toString();
416                     }
417                     processEmbed(curi, res, element); // TODO: include attribute too
418                 }
419             } catch (URIException e) {
420                 curi.addLocalizedError(getName(), e, "BAD CODEBASE " + codebase);
421             } catch (IllegalArgumentException e) {
422                 DevUtils.logger.log(Level.WARNING, "processGeneralTag()\n" +
423                     "codebase=" + codebase + " res=" + res + "\n" +
424                     DevUtils.extraInfo(), e);
425             }
426         }
427         
428         // finish handling form action, now method is available
429         if(action != null) {
430             if(method == null || "GET".equalsIgnoreCase(method.toString()) 
431                     || ! ((Boolean)getUncheckedAttribute(curi,
432                             ATTR_EXTRACT_ONLY_FORM_GETS)).booleanValue()) {
433                 processLink(curi, action, actionContext);
434             }
435         }
436         
437         // finish handling VALUE
438         if(valueVal != null) {
439             if("PARAM".equalsIgnoreCase(elementStr) && "flashvars".equalsIgnoreCase(nameVal.toString())) {
440                 // special handling for <PARAM NAME='flashvars" VALUE="">
441                 String queryStringLike = valueVal.toString();
442                 // treat value as query-string-like "key=value[;key=value]*" pairings
443                 considerQueryStringValues(curi, queryStringLike, valueContext,Link.SPECULATIVE_HOP);
444             } else {
445                 // regular VALUE handling
446                 if (extractValueAttributes) {
447                     considerIfLikelyUri(curi,valueVal,valueContext,Link.NAVLINK_HOP);
448                 }
449             }
450         }
451     }
452 
453     /***
454      * Consider a query-string-like collections of key=value[;key=value]
455      * pairs for URI-like strings in the values. Where URI-like strings are
456      * found, add as discovered outlink. 
457      * 
458      * @param curi origin CrawlURI
459      * @param queryString query-string-like string
460      * @param valueContext page context where found
461      */
462     protected void considerQueryStringValues(CrawlURI curi,
463             CharSequence queryString, CharSequence valueContext, char hopType) {
464         for(String pairString : queryString.toString().split(";")) {
465             String[] keyVal = pairString.split("=");
466             if(keyVal.length==2) {
467                 considerIfLikelyUri(curi,keyVal[1],valueContext, hopType);
468             }
469         }
470     }
471 
472     /***
473      * Consider whether a given string is URI-like. If so, add as discovered 
474      * outlink. 
475      * 
476      * @param curi origin CrawlURI
477      * @param queryString query-string-like string
478      * @param valueContext page context where found
479 
480      */
481     protected void considerIfLikelyUri(CrawlURI curi, CharSequence candidate, 
482             CharSequence valueContext, char hopType) {
483         if(UriUtils.isLikelyUriHtmlContextLegacy(candidate)) {
484             addLinkFromString(curi,candidate,valueContext,hopType);
485         }
486     }

501     static final String JAVASCRIPT = "(?i)^javascript:.*";

503     /***
504      * Handle generic HREF cases.
505      * 
506      * @param curi
507      * @param value
508      * @param context
509      */
510     protected void processLink(CrawlURI curi, final CharSequence value,
511             CharSequence context) {
512         if (TextUtils.matches(JAVASCRIPT, value)) {
513             processScriptCode(curi, value. subSequence(11, value.length()));
514         } else {    
515             if (logger.isLoggable(Level.FINEST)) {
516                 logger.finest("link: " + value.toString() + " from " + curi);
517             }
518             addLinkFromString(curi, value, context, Link.NAVLINK_HOP);
519             this.numberOfLinksExtracted++;
520         }
521     }

523     protected void addLinkFromString(CrawlURI curi, CharSequence uri,
524             CharSequence context, char hopType) {
525         try {
526             // We do a 'toString' on context because its a sequence from
527             // the underlying ReplayCharSequence and the link its about
528             // to become a part of is expected to outlive the current
529             // ReplayCharSequence.
530             curi.createAndAddLinkRelativeToBase(uri.toString(), context.toString(),
531                 hopType);
532         } catch (URIException e) {
533             if (getController() != null) {
534                 getController().logUriError(e, curi.getUURI(), uri);
535             } else {
536                 logger.info("Failed createAndAddLinkRelativeToBase " +
537                     curi + ", " + uri + ", " + context + ", " + hopType +
538                     ": " + e);
539             }
540         }
541     }

544             CharSequence context) {
545         processEmbed(curi, value, context, Link.EMBED_HOP);

548     protected void processEmbed(CrawlURI curi, final CharSequence value,
549             CharSequence context, char hopType) {
550         if (logger.isLoggable(Level.FINEST)) {
551             logger.finest("embed (" + hopType + "): " + value.toString() +
552                 " from " + curi);
553         }
554         addLinkFromString(curi,
555             (value instanceof String)?
556                 (String)value: value.toString(),
557             context, hopType);
558         this.numberOfLinksExtracted++;
559     }

561     public void extract(CrawlURI curi) {
562         if (!isHttpTransactionContentToProcess(curi) ||
563                 ! (isExpectedMimeType(curi.getContentType(), "text/html")
564                    || isExpectedMimeType(curi.getContentType(), "application/xhtml")
565                    || isExpectedMimeType(curi.getContentType(), "text/vnd.wap.wml")
566                    || isExpectedMimeType(curi.getContentType(), "application/vnd.wap.wml")
567                    || isExpectedMimeType(curi.getContentType(), "application/vnd.wap.xhtml"))) {
568             return;
569         }
570 
571         final boolean ignoreUnexpectedHTML =
572              ((Boolean)getUncheckedAttribute(curi, 
573                  ATTR_IGNORE_UNEXPECTED_HTML)).booleanValue();        
574 
575         if (ignoreUnexpectedHTML) {
576             try {
577                 if(!isHtmlExpectedHere(curi)) {
578                     // HTML was not expected (eg a GIF was expected) so ignore
579                     // (as if a soft 404)
580                     return;
581                 }
582             } catch (URIException e) {
583                 logger.severe("Failed expectedHTML test: " + e.getMessage());
584             }
585         }
586 
587         this.numberOfCURIsHandled++;
588 
589         ReplayCharSequence cs = null;
590         
591         try {
592            HttpRecorder hr = curi.getHttpRecorder();
593            if (hr == null) {
594                throw new IOException("Why is recorder null here?");
595            }
596            cs = hr.getReplayCharSequence();
597         } catch (IOException e) {
598             curi.addLocalizedError(this.getName(), e,
599                 "Failed get of replay char sequence " + curi.toString() +
600                     " " + e.getMessage());
601             logger.log(Level.SEVERE,"Failed get of replay char sequence in " +
602                 Thread.currentThread().getName(), e);
603         }
604         
605         if (cs == null) {
606             return;
607         }
608 
609         // We have a ReplayCharSequence open.  Wrap all in finally so we
610         // for sure close it before we leave.
611         try {
612             // Extract all links from the charsequence
613             extract(curi, cs);
614             // Set flag to indicate that link extraction is completed.
615             curi.linkExtractorFinished();
616         } finally {
617             if (cs != null) {
618                 try {
619                     cs.close();
620                 } catch (IOException ioe) {
621                     logger.warning(TextUtils.exceptionToString(
622                         "Failed close of ReplayCharSequence.", ioe));
623                 }
624             }
625         }
626     }

628     /***
629      * Run extractor.
630      * This method is package visible to ease testing.
631      * @param curi CrawlURI we're processing.
632      * @param cs Sequence from underlying ReplayCharSequence. This
633      * is TRANSIENT data. Make a copy if you want the data to live outside
634      * of this extractors' lifetime.
635      */
636     void extract(CrawlURI curi, CharSequence cs) {
637         Matcher tags = TextUtils.getMatcher(RELEVANT_TAG_EXTRACTOR, cs);
638         while(tags.find()) {
639             if(Thread.interrupted()){
640                 break;
641             }
642             if (tags.start(8) > 0) {
643                 // comment match
644                 // for now do nothing
645             } else if (tags.start(7) > 0) {
646                 // <meta> match
647                 int start = tags.start(5);
648                 int end = tags.end(5);
649                 assert start >= 0: "Start is: " + start + ", " + curi;
650                 assert end >= 0: "End is :" + end + ", " + curi;
651                 if (processMeta(curi,
652                     cs.subSequence(start, end))) {
653 
654                     // meta tag included NOFOLLOW; abort processing
655                     break;
656                 }
657             } else if (tags.start(5) > 0) {
658                 // generic <whatever> match
659                 int start5 = tags.start(5);
660                 int end5 = tags.end(5);
661                 assert start5 >= 0: "Start is: " + start5 + ", " + curi;
662                 assert end5 >= 0: "End is :" + end5 + ", " + curi;
663                 int start6 = tags.start(6);
664                 int end6 = tags.end(6);
665                 assert start6 >= 0: "Start is: " + start6 + ", " + curi;
666                 assert end6 >= 0: "End is :" + end6 + ", " + curi;
667                 processGeneralTag(curi,
668                     cs.subSequence(start6, end6),
669                     cs.subSequence(start5, end5));
670 
671             } else if (tags.start(1) > 0) {
672                 // <script> match
673                 int start = tags.start(1);
674                 int end = tags.end(1);
675                 assert start >= 0: "Start is: " + start + ", " + curi;
676                 assert end >= 0: "End is :" + end + ", " + curi;
677                 assert tags.end(2) >= 0: "Tags.end(2) illegal " + tags.end(2) +
678                     ", " + curi;
679                 processScript(curi, cs.subSequence(start, end),
680                     tags.end(2) - start);
681 
682             } else if (tags.start(3) > 0){
683                 // <style... match
684                 int start = tags.start(3);
685                 int end = tags.end(3);
686                 assert start >= 0: "Start is: " + start + ", " + curi;
687                 assert end >= 0: "End is :" + end + ", " + curi;
688                 assert tags.end(4) >= 0: "Tags.end(4) illegal " + tags.end(4) +
689                     ", " + curi;
690                 processStyle(curi, cs.subSequence(start, end),
691                     tags.end(4) - start);
692             }
693         }
694         TextUtils.recycleMatcher(tags);
695     }

702     /***
703      * Test whether this HTML is so unexpected (eg in place of a GIF URI)
704      * that it shouldn't be scanned for links.
705      *
706      * @param curi CrawlURI to examine.
707      * @return True if HTML is acceptable/expected here
708      * @throws URIException
709      */
710     protected boolean isHtmlExpectedHere(CrawlURI curi) throws URIException {
711         String path = curi.getUURI().getPath();
712         if(path==null) {
713             // no path extension, HTML is fine
714             return true;
715         }
716         int dot = path.lastIndexOf('.');
717         if (dot < 0) {
718             // no path extension, HTML is fine
719             return true;
720         }
721         if(dot<(path.length()-5)) {
722             // extension too long to recognize, HTML is fine
723             return true;
724         }
725         String ext = path.substring(dot+1);
726         return ! TextUtils.matches(NON_HTML_PATH_EXTENSION, ext);
727     }

742     /***
743      * Process metadata tags.
744      * @param curi CrawlURI we're processing.
745      * @param cs Sequence from underlying ReplayCharSequence. This
746      * is TRANSIENT data. Make a copy if you want the data to live outside
747      * of this extractors' lifetime.
748      * @return True robots exclusion metatag.
749      */
750     protected boolean processMeta(CrawlURI curi, CharSequence cs) {
751         Matcher attr = TextUtils.getMatcher(EACH_ATTRIBUTE_EXTRACTOR, cs);
752         String name = null;
753         String httpEquiv = null;
754         String content = null;
755         while (attr.find()) {
756             int valueGroup =
757                 (attr.start(14) > -1) ? 14 : (attr.start(15) > -1) ? 15 : 16;
758             CharSequence value =
759                 cs.subSequence(attr.start(valueGroup), attr.end(valueGroup));
760             value = TextUtils.unescapeHtml(value);
761             if (attr.group(1).equalsIgnoreCase("name")) {
762                 name = value.toString();
763             } else if (attr.group(1).equalsIgnoreCase("http-equiv")) {
764                 httpEquiv = value.toString();
765             } else if (attr.group(1).equalsIgnoreCase("content")) {
766                 content = value.toString();
767             }
768             // TODO: handle other stuff
769         }
770         TextUtils.recycleMatcher(attr);
771 
772         // Look for the 'robots' meta-tag
773         if("robots".equalsIgnoreCase(name) && content != null ) {
774             curi.putString(A_META_ROBOTS, content);
775             RobotsHonoringPolicy policy =
776                 getSettingsHandler().getOrder().getRobotsHonoringPolicy();
777             String contentLower = content.toLowerCase();
778             if ((policy == null
779                 || (!policy.isType(curi, RobotsHonoringPolicy.IGNORE)
780                     && !policy.isType(curi, RobotsHonoringPolicy.CUSTOM)))
781                 && (contentLower.indexOf("nofollow") >= 0
782                     || contentLower.indexOf("none") >= 0)) {
783                 // if 'nofollow' or 'none' is specified and the
784                 // honoring policy is not IGNORE or CUSTOM, end html extraction
785                 logger.fine("HTML extraction skipped due to robots meta-tag for: "
786                                 + curi.toString());
787                 return true;
788             }
789         } else if ("refresh".equalsIgnoreCase(httpEquiv) && content != null) {
790             int urlIndex = content.indexOf("=") + 1;
791             if(urlIndex>0) {
792                 String refreshUri = content.substring(urlIndex);
793                 try {
794                     curi.createAndAddLinkRelativeToBase(refreshUri, "meta",
795                         Link.REFER_HOP);
796                 } catch (URIException e) {
797                     if (getController() != null) {
798                         getController().logUriError(e, curi.getUURI(), refreshUri);
799                     } else {
800                         logger.info("Failed createAndAddLinkRelativeToBase " +
801                             curi + ", " + cs + ", " + refreshUri + ": " + e);
802                     }
803                 }
804             }
805         }
806         return false;
807     }
