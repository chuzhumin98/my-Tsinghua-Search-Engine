420     protected void innerProcess(final CrawlURI curi)
421     throws InterruptedException {
422         if (!canFetch(curi)) {
423             // Cannot fetch this, due to protocol, retries, or other problems
424             return;
425         }
426 
427         this.curisHandled++;
428 
429         // Note begin time
430         curi.putLong(A_FETCH_BEGAN_TIME, System.currentTimeMillis());
431 
432         // Get a reference to the HttpRecorder that is set into this ToeThread.
433         HttpRecorder rec = HttpRecorder.getHttpRecorder();
434         
435         // Shall we get a digest on the content downloaded?
436         boolean digestContent  = ((Boolean)getUncheckedAttribute(curi,
437                 ATTR_DIGEST_CONTENT)).booleanValue();
438         String algorithm = null;
439         if (digestContent) {
440             algorithm = ((String)getUncheckedAttribute(curi,
441                 ATTR_DIGEST_ALGORITHM));
442             rec.getRecordedInput().setDigest(algorithm);
443         } else {
444             // clear
445             rec.getRecordedInput().setDigest((MessageDigest)null);
446         }        
447         
448         // Below we do two inner classes that add check of midfetch
449         // filters just as we're about to receive the response body.
450         String curiString = curi.getUURI().toString();
451         HttpMethodBase method = null;
452         if (curi.isPost()) {
453             method = new HttpRecorderPostMethod(curiString, rec) {
454                 protected void readResponseBody(HttpState state,
455                         HttpConnection conn)
456                 throws IOException, HttpException {
457                     addResponseContent(this, curi);
458                     if (checkMidfetchAbort(curi, this.httpRecorderMethod, conn)) {
459                         doAbort(curi, this, MIDFETCH_ABORT_LOG);
460                     } else {
461                         super.readResponseBody(state, conn);
462                     }
463                 }
464             };
465         } else {
466             method = new HttpRecorderGetMethod(curiString, rec) {
467                 protected void readResponseBody(HttpState state,
468                         HttpConnection conn)
469                 throws IOException, HttpException {
470                     addResponseContent(this, curi);
471                     if (checkMidfetchAbort(curi, this.httpRecorderMethod,
472                             conn)) {
473                         doAbort(curi, this, MIDFETCH_ABORT_LOG);
474                     } else {
475                         super.readResponseBody(state, conn);
476                     }
477                 }
478             };
479         }
480 
481         HostConfiguration customConfigOrNull = configureMethod(curi, method);
482         
483         // Set httpRecorder into curi. Subsequent code both here and later
484         // in extractors expects to find the HttpRecorder in the CrawlURI.
485         curi.setHttpRecorder(rec);
486         
487         // Populate credentials. Set config so auth. is not automatic.
488         boolean addedCredentials = populateCredentials(curi, method);
489         method.setDoAuthentication(addedCredentials);
490         
491         // set hardMax on bytes (if set by operator)
492         long hardMax = getMaxLength(curi);
493         // set overall timeout (if set by operator)
494         long timeoutMs = 1000 * getTimeout(curi);
495         // Get max fetch rate (bytes/ms). It comes in in KB/sec
496         long maxRateKBps = getMaxFetchRate(curi);
497         rec.getRecordedInput().setLimits(hardMax, timeoutMs, maxRateKBps);
498         
499         try {
500             this.http.executeMethod(customConfigOrNull, method);
501         } catch (RecorderTooMuchHeaderException ex) {
502             // when too much header material, abort like other truncations
503             doAbort(curi, method, HEADER_TRUNC);
504         } catch (IOException e) {
505         	failedExecuteCleanup(method, curi, e);
506         	return;
507         } catch (ArrayIndexOutOfBoundsException e) {
508             // For weird windows-only ArrayIndex exceptions in native
509             // code... see
510             // http://forum.java.sun.com/thread.jsp?forum=11&thread=378356
511             // treating as if it were an IOException
512             failedExecuteCleanup(method, curi, e);
513             return;
514         }
515         
516         // set softMax on bytes to get (if implied by content-length) 
517         long softMax = method.getResponseContentLength();
518 
519         try {
520             if (!method.isAborted()) {
521                 // Force read-to-end, so that any socket hangs occur here,
522                 // not in later modules.
523                 rec.getRecordedInput().readFullyOrUntil(softMax);
524             }
525         } catch (RecorderTimeoutException ex) {
526             doAbort(curi, method, TIMER_TRUNC);
527         } catch (RecorderLengthExceededException ex) {
528             doAbort(curi, method, LENGTH_TRUNC);
529         } catch (IOException e) {
530             cleanup(curi, e, "readFully", S_CONNECT_LOST);
531             return;
532         } catch (ArrayIndexOutOfBoundsException e) {
533             // For weird windows-only ArrayIndex exceptions from native code
534             // see http://forum.java.sun.com/thread.jsp?forum=11&thread=378356
535             // treating as if it were an IOException
536             cleanup(curi, e, "readFully", S_CONNECT_LOST);
537             return;
538         } finally {
539             // ensure recording has stopped
540             rec.closeRecorders();
541             if (!method.isAborted()) {
542                 method.releaseConnection();
543             }
544             // Note completion time
545             curi.putLong(A_FETCH_COMPLETED_TIME, System.currentTimeMillis());
546             // Set the response charset into the HttpRecord if available.
547             setCharacterEncoding(rec, method);
548             setSizes(curi, rec);
549         }
550  
551         if (digestContent) {
552             curi.setContentDigest(algorithm,
553                 rec.getRecordedInput().getDigestValue());
554         }
555         if (logger.isLoggable(Level.INFO)) {
556             logger.info((curi.isPost()? "POST": "GET") + " " +
557                 curi.getUURI().toString() + " " + method.getStatusCode() +
558                 " " + rec.getRecordedInput().getSize() + " " +
559                 curi.getContentType());
560         }
561 
562         if (curi.isSuccess() && addedCredentials) {
563             // Promote the credentials from the CrawlURI to the CrawlServer
564             // so they are available for all subsequent CrawlURIs on this
565             // server.
566             promoteCredentials(curi);
567             if (logger.isLoggable(Level.FINE)) {
568                 // Print out the cookie.  Might help with the debugging.
569                 Header setCookie = method.getResponseHeader("set-cookie");
570                 if (setCookie != null) {
571                     logger.fine(setCookie.toString().trim());
572                 }
573             }
574         } else if (method.getStatusCode() == HttpStatus.SC_UNAUTHORIZED) {
575             // 401 is not 'success'.
576             handle401(method, curi);
577         }
578         
579         if (rec.getRecordedInput().isOpen()) {
580             logger.severe(curi.toString() + " RIS still open. Should have" +
581                 " been closed by method release: " +
582                 Thread.currentThread().getName());
583             try {
584                 rec.getRecordedInput().close();
585             } catch (IOException e) {
586                 logger.log(Level.SEVERE,"second-chance RIS close failed",e);
587             }
588         }
589     }

719      * @param curi

739          }

751         // Don't auto-follow redirects

753         
754 //        // set soTimeout
755 //        method.getParams().setSoTimeout(
756 //                ((Integer) getUncheckedAttribute(curi, ATTR_SOTIMEOUT_MS))

782         if(maxLength > 0 &&

798                                 curi.getViaContext().toString()))) {

803                 !(via.startsWith(HTTPS_SCHEME) &&
804                     curi.getUURI().getScheme().equals(HTTP_SCHEME))) {
805                 method.setRequestHeader(REFERER, via);
806             }
807         }
808         
809         if(!curi.isPrerequisite()) {
810             setConditionalGetHeader(curi, method, ATTR_SEND_IF_MODIFIED_SINCE, 
811                     CoreAttributeConstants.A_LAST_MODIFIED_HEADER, "If-Modified-Since");
812             setConditionalGetHeader(curi, method, ATTR_SEND_IF_NONE_MATCH, 
813                     CoreAttributeConstants.A_ETAG_HEADER, "If-None-Match");
814         }
815         
816         // TODO: What happens if below method adds a header already
817         // added above: e.g. Connection, Range, or Referer?
818         setAcceptHeaders(curi, method);
819         
820         HostConfiguration config = new HostConfiguration(http.getHostConfiguration());
821         configureProxy(curi, config);
822         configureBindAddress(curi, config);
823         return config;
824     }

909     /***
910      * Add credentials if any to passed <code>method</code>.
911      *
912      * Do credential handling.  Credentials are in two places.  1. Credentials
913      * that succeeded are added to the CrawlServer (Or rather, avatars for
914      * credentials are whats added because its not safe to keep around
915      * references to credentials).  2. Credentials to be tried are in the curi.
916      * Returns true if found credentials to be tried.
917      *
918      * @param curi Current CrawlURI.
919      * @param method The method to add to.
920      * @return True if prepopulated <code>method</code> with credentials AND the
921      * credentials came from the <code>curi</code>, not from the CrawlServer.
922      * The former is  special in that if the <code>curi</curi> credentials
923      * succeed, then the caller needs to promote them from the CrawlURI to the
924      * CrawlServer so they are available for all subsequent CrawlURIs on this
925      * server.
926      */
927     private boolean populateCredentials(CrawlURI curi, HttpMethod method) {
928         // First look at the server avatars. Add any that are to be volunteered
929         // on every request (e.g. RFC2617 credentials).  Every time creds will
930         // return true when we call 'isEveryTime().
931         CrawlServer server =
932             getController().getServerCache().getServerFor(curi);
933         if (server.hasCredentialAvatars()) {
934             Set avatars = server.getCredentialAvatars();
935             for (Iterator i = avatars.iterator(); i.hasNext();) {
936                 CredentialAvatar ca = (CredentialAvatar)i.next();
937                 Credential c = ca.getCredential(getSettingsHandler(), curi);
938                 if (c.isEveryTime()) {
939                     c.populate(curi, this.http, method, ca.getPayload());
940                 }
941             }
942         }
943 
944         boolean result = false;
945 
946         // Now look in the curi.  The Curi will have credentials loaded either
947         // by the handle401 method if its a rfc2617 or it'll have been set into
948         // the curi by the preconditionenforcer as this login uri came through.
949         if (curi.hasCredentialAvatars()) {
950             Set avatars = curi.getCredentialAvatars();
951             for (Iterator i = avatars.iterator(); i.hasNext();) {
952                 CredentialAvatar ca = (CredentialAvatar)i.next();
953                 Credential c = ca.getCredential(getSettingsHandler(), curi);
954                 if (c.populate(curi, this.http, method, ca.getPayload())) {
955                     result = true;
956                 }
957             }
958         }
959 
960         return result;
961     }

963     /***
964      * Promote successful credential to the server.
965      *
966      * @param curi CrawlURI whose credentials we are to promote.
967      */
968     private void promoteCredentials(final CrawlURI curi) {
969         if (!curi.hasCredentialAvatars()) {
970             logger.severe("No credentials to promote when there should be " +
971                 curi);
972         } else {
973             Set avatars = curi.getCredentialAvatars();
974             for (Iterator i = avatars.iterator(); i.hasNext();) {
975                 CredentialAvatar ca = (CredentialAvatar)i.next();
976                 curi.removeCredentialAvatar(ca);
977                 // The server to attach too may not be the server that hosts
978                 // this passed curi.  It might be of another subdomain.
979                 // The avatar needs to be added to the server that is dependent
980                 // on this precondition.  Find it by name.  Get the name from
981                 // the credential this avatar represents.
982                 Credential c = ca.getCredential(getSettingsHandler(), curi);
983                 String cd = null;
984                 try {
985                     cd = c.getCredentialDomain(curi);
986                 }
987                 catch (AttributeNotFoundException e) {
988                     logger.severe("Failed to get cred domain for " + curi +
989                         " for " + ca + ": " + e.getMessage());
990                 }
991                 if (cd != null) {
992                     CrawlServer cs
993                         = getController().getServerCache().getServerFor(cd);
994                     if (cs != null) {
995                         cs.addCredentialAvatar(ca);
996                     }
997                 }
998             }
999         }

1002     /***
1003      * Server is looking for basic/digest auth credentials (RFC2617). If we have
1004      * any, put them into the CrawlURI and have it come around again. Presence
1005      * of the credential serves as flag to frontier to requeue promptly. If we
1006      * already tried this domain and still got a 401, then our credentials are
1007      * bad. Remove them and let this curi die.
1008      *
1009      * @param method Method that got a 401.
1010      * @param curi CrawlURI that got a 401.
1011      */
1012     protected void handle401(final HttpMethod method, final CrawlURI curi) {
1013         AuthScheme authscheme = getAuthScheme(method, curi);
1014         if (authscheme == null) {
1015         	return;
1016         }
1017         String realm = authscheme.getRealm();
1018         
1019         // Look to see if this curi had rfc2617 avatars loaded.  If so, are
1020         // any of them for this realm?  If so, then the credential failed
1021         // if we got a 401 and it should be let die a natural 401 death.
1022         Set curiRfc2617Credentials = getCredentials(getSettingsHandler(),
1023         		curi, Rfc2617Credential.class);
1024         Rfc2617Credential extant = Rfc2617Credential.
1025 		    getByRealm(curiRfc2617Credentials, realm, curi);
1026         if (extant != null) {
1027         	// Then, already tried this credential.  Remove ANY rfc2617
1028         	// credential since presence of a rfc2617 credential serves
1029         	// as flag to frontier to requeue this curi and let the curi
1030         	// die a natural death.
1031         	extant.detachAll(curi);
1032         	logger.warning("Auth failed (401) though supplied realm " +
1033         			realm + " to " + curi.toString());
1034         } else {
1035         	// Look see if we have a credential that corresponds to this
1036         	// realm in credential store.  Filter by type and credential
1037         	// domain.  If not, let this curi die. Else, add it to the
1038         	// curi and let it come around again. Add in the AuthScheme
1039         	// we got too.  Its needed when we go to run the Auth on
1040         	// second time around.
1041         	CredentialStore cs =
1042         		CredentialStore.getCredentialStore(getSettingsHandler());
1043         	if (cs == null) {
1044         		logger.severe("No credential store for " + curi);
1045         	} else {
1046                 CrawlServer server = getController().getServerCache().
1047                     getServerFor(curi);
1048         		Set storeRfc2617Credentials = cs.subset(curi,
1049         		    Rfc2617Credential.class, server.getName());
1050         		if (storeRfc2617Credentials == null ||
1051         				storeRfc2617Credentials.size() <= 0) {
1052         			logger.info("No rfc2617 credentials for " + curi);
1053         		} else {
1054         			Rfc2617Credential found = Rfc2617Credential.
1055 					    getByRealm(storeRfc2617Credentials, realm, curi);
1056         			if (found == null) {
1057         				logger.info("No rfc2617 credentials for realm " +
1058         						realm + " in " + curi);
1059         			} else {
1060         				found.attach(curi, authscheme.getRealm());
1061         				logger.info("Found credential for realm " + realm +
1062         				    " in store for " + curi.toString());
1063         			}
1064         		}
1065         	}
1066         }
1067     }

1069     /***
1070      * @param method Method that got a 401.
1071      * @param curi CrawlURI that got a 401.
1072      * @return Returns first wholesome authscheme found else null.
1073      */
1074     protected AuthScheme getAuthScheme(final HttpMethod method,
1075             final CrawlURI curi) {
1076         Header [] headers = method.getResponseHeaders("WWW-Authenticate");
1077         if (headers == null || headers.length <= 0) {
1078             logger.info("We got a 401 but no WWW-Authenticate challenge: " +
1079                 curi.toString());
1080             return null;
1081         }
1082 
1083         Map authschemes = null;
1084         try {
1085             authschemes = AuthChallengeParser.parseChallenges(headers);
1086         } catch(MalformedChallengeException e) {
1087             logger.info("Failed challenge parse: " + e.getMessage());
1088         }
1089         if (authschemes == null || authschemes.size() <= 0) {
1090             logger.info("We got a 401 and WWW-Authenticate challenge" +
1091                 " but failed parse of the header " + curi.toString());
1092             return null;
1093         }            
1094          
1095         AuthScheme result = null;
1096         // Use the first auth found.
1097         for (Iterator i = authschemes.keySet().iterator();
1098                 result == null && i.hasNext();) {
1099         	String key = (String)i.next();
1100             String challenge = (String)authschemes.get(key);
1101             if (key == null || key.length() <= 0 || challenge == null ||
1102                   challenge.length() <= 0) {
1103             	logger.warning("Empty scheme: " + curi.toString() +
1104                   ": " + headers);
1105             }
1106         	AuthScheme authscheme = null;
1107         	if (key.equals("basic")) {
1108         		authscheme = new BasicScheme();
1109         	} else if (key.equals("digest")) {
1110         		authscheme = new DigestScheme();
1111         	} else {
1112         		logger.info("Unsupported scheme: " + key);
1113         		continue;
1114         	}
1115             
1116             try {
1117 				authscheme.processChallenge(challenge);
1118 			} catch (MalformedChallengeException e) {
1119 				logger.info(e.getMessage() + " " + curi + " " + headers);
1120                 continue;
1121 			}
1122         	if (authscheme.isConnectionBased()) {
1123         		logger.info("Connection based " + authscheme);
1124         		continue;
1125         	}
1126         	
1127         	if (authscheme.getRealm() == null ||
1128         			authscheme.getRealm().length() <= 0) {
1129         		logger.info("Empty realm " + authscheme + " for " + curi);
1130         		continue;
1131         	}
1132         	result = authscheme;
1133         }
1134         
1135         return result;
1136     }

1138     /***
1139      * @param handler Settings Handler.
1140      * @param curi CrawlURI that got a 401.
1141      * @param type Class of credential to get from curi.
1142      * @return Set of credentials attached to this curi.
1143      */
1144     private Set<Credential> getCredentials(SettingsHandler handler, 
1145             CrawlURI curi, Class type) {

1147 
1148         if (curi.hasCredentialAvatars()) {
1149             for (Iterator i = curi.getCredentialAvatars().iterator();
1150                     i.hasNext();) {
1151                 CredentialAvatar ca = (CredentialAvatar)i.next();

1153                     if (result == null) {
1154                         result = new HashSet<Credential>();
1155                     }
1156                     result.add(ca.getCredential(handler, curi));
1157                 }
1158             }
1159         }
1160         return result;
1161     }
