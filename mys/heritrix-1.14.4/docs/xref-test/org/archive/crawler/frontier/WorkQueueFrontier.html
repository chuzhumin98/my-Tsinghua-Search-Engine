80          public void run() {

278     /***
279      * Initializes the Frontier, given the supplied CrawlController.
280      *
281      * @see org.archive.crawler.framework.Frontier#initialize(org.archive.crawler.framework.CrawlController)
282      */
283     public void initialize(CrawlController c)
284             throws FatalConfigurationException, IOException {
285         // Call the super method. It sets up frontier journalling.
286         super.initialize(c);
287         this.controller = c;
288         
289         initQueuesOfQueues();
290         
291         this.targetSizeForReadyQueues = (Integer)getUncheckedAttribute(null,
292             ATTR_TARGET_READY_QUEUES_BACKLOG);
293         if (this.targetSizeForReadyQueues < 1) {
294             this.targetSizeForReadyQueues = 1;
295         }
296         this.wakeTimer = new Timer("waker for " + c.toString());
297         
298         try {
299             if (workQueueDataOnDisk()
300                     && getQueueAssignmentPolicy(null).maximumNumberOfKeys() >= 0
301                     && getQueueAssignmentPolicy(null).maximumNumberOfKeys() <= 
302                         MAX_QUEUES_TO_HOLD_ALLQUEUES_IN_MEMORY) {
303                 this.allQueues = new ObjectIdentityMemCache<WorkQueue>(701, .9f, 100);
304             } else {
305                 this.allQueues = c.getBigMap("allqueues", WorkQueue.class);
306                 if (logger.isLoggable(Level.FINE)) {
307                     Iterator<String> i = this.allQueues.keySet().iterator();
308                     try {
309                         for (; i.hasNext();) {
310                             logger.fine((String) i.next());
311                         }
312                     } finally {
313                         StoredIterator.close(i);
314                     }
315                 }
316             }
317             this.alreadyIncluded = createAlreadyIncluded();
318             initQueue();
319         } catch (IOException e) {
320             e.printStackTrace();
321             throw (FatalConfigurationException)
322                 new FatalConfigurationException(e.getMessage()).initCause(e);
323         } catch (Exception e) {
324             e.printStackTrace();
325             throw (FatalConfigurationException)
326                 new FatalConfigurationException(e.getMessage()).initCause(e);
327         }
328         
329         initCostPolicy();
330         
331         loadSeeds();
332     }

334     /***
335      * Set up the various queues-of-queues used by the frontier. Override
336      * in implementing subclasses to reduce or eliminate risk of queues
337      * growing without bound. 
338      */
339     protected void initQueuesOfQueues() {
340         // small risk of OutOfMemoryError: if 'hold-queues' is false,
341         // readyClassQueues may grow in size without bound
342         readyClassQueues = new LinkedBlockingQueue<String>();
343         // risk of OutOfMemoryError: in large crawls, 
344         // inactiveQueues may grow in size without bound
345         inactiveQueues = new LinkedBlockingQueue<String>();
346         // risk of OutOfMemoryError: in large crawls with queue max-budgets, 
347         // inactiveQueues may grow in size without bound
348         retiredQueues = new LinkedBlockingQueue<String>();
349         // small risk of OutOfMemoryError: in large crawls with many 
350         // unresponsive queues, an unbounded number of snoozed queues 
351         // may exist
352         snoozedClassQueues = Collections.synchronizedSortedSet(new TreeSet<WorkQueue>());
353     }

371     /* (non-Javadoc)
372      * @see org.archive.crawler.frontier.AbstractFrontier#crawlEnded(java.lang.String)
373      */
374     public void crawlEnded(String sExitMessage) {
375         // Cleanup.  CrawlJobs persist after crawl has finished so undo any
376         // references.
377         if (this.alreadyIncluded != null) {
378             this.alreadyIncluded.close();
379             this.alreadyIncluded = null;
380         }
381         
382         try {
383             closeQueue();
384         } catch (IOException e) {
385             // FIXME exception handling
386             e.printStackTrace();
387         }
388         this.wakeTimer.cancel();
389         
390         this.allQueues.close();
391         this.allQueues = null;
392         this.inProcessQueues = null;
393         this.readyClassQueues = null;
394         this.snoozedClassQueues = null;
395         this.inactiveQueues = null;
396         this.retiredQueues = null;
397         
398         this.costAssignmentPolicy = null;
399         
400         // Clearing controller is a problem. We get NPEs in #preNext.
401         super.crawlEnded(sExitMessage);
402         this.controller = null;
403     }

421         // Canonicalization may set forceFetch flag.  See

423         String canon = canonicalize(caUri);
424         if (caUri.forceFetch()) {

426         } else {
427             alreadyIncluded.add(canon, caUri);
428         }

458     /***
459      * Send a CrawlURI to the appropriate subqueue.
460      * 
461      * @param curi
462      */
463     protected void sendToQueue(CrawlURI curi) {
464         WorkQueue wq = getQueueFor(curi);
465         synchronized (wq) {
466             wq.enqueue(this, curi);
467             if(!wq.isRetired()) {
468                 incrementQueuedUriCount();
469             }
470             if(!wq.isHeld()) {
471                 wq.setHeld();
472                 if(holdQueues() && readyClassQueues.size()>=targetSizeForReadyQueues()) {
473                     deactivateQueue(wq);
474                 } else {
475                     replenishSessionBalance(wq);
476                     readyQueue(wq);
477                 }
478             }
479             WorkQueue laq = longestActiveQueue;
480             if(!wq.isRetired()&&((laq==null) || wq.getCount() > laq.getCount())) {
481                 longestActiveQueue = wq; 
482             }
483         }
484     }

548     /*** 
549      * Accomodate any changes in settings.
550      * 
551      * @see org.archive.crawler.framework.Frontier#kickUpdate()
552      */
553     public void kickUpdate() {
554         super.kickUpdate();
555         int target = (Integer)getUncheckedAttribute(null,
556                 ATTR_TARGET_READY_QUEUES_BACKLOG);
557         if (target < 1) {
558             target = 1;
559         }
560         this.targetSizeForReadyQueues = target; 
561         try {
562             initCostPolicy();
563         } catch (FatalConfigurationException fce) {
564             throw new RuntimeException(fce);
565         }
566         // The rules for a 'retired' queue may have changed; so,
567         // unretire all queues to 'inactive'. If they still qualify
568         // as retired/overbudget next time they come up, they'll
569         // be re-retired; if not, they'll get a chance to become
570         // active under the new rules.
571         Object key = this.retiredQueues.poll();
572         while (key != null) {
573             WorkQueue q = (WorkQueue)this.allQueues.get((String)key);
574             if(q != null) {
575                 unretireQueue(q);
576             }
577             key = this.retiredQueues.poll();
578         }
579     }
580     /***
581      * Restore a retired queue to the 'inactive' state. 
582      * 
583      * @param q
584      */
585     private void unretireQueue(WorkQueue q) {
586         deactivateQueue(q);
587         q.setRetired(false); 
588         incrementQueuedUriCount(q.getCount());
589     }

610     /***
611      * Return the next CrawlURI to be processed (and presumably
612      * visited/fetched) by a a worker thread.
613      *
614      * Relies on the readyClassQueues having been loaded with
615      * any work queues that are eligible to provide a URI. 
616      *
617      * @return next CrawlURI to be processed. Or null if none is available.
618      *
619      * @see org.archive.crawler.framework.Frontier#next()
620      */
621     public CrawlURI next()
622     throws InterruptedException, EndedException {
623         while (true) { // loop left only by explicit return or exception
624             long now = System.currentTimeMillis();
625 
626             // Do common checks for pause, terminate, bandwidth-hold
627             preNext(now);
628             
629             // allow up-to-1 thread to fill readyClassQueues to target
630             if(readyFiller.tryAcquire()) {
631                 try {
632                     int activationsNeeded = targetSizeForReadyQueues() - readyClassQueues.size();
633                     while(activationsNeeded > 0 && !inactiveQueues.isEmpty()) {
634                         activateInactiveQueue();
635                         activationsNeeded--;
636                     }
637                 } finally {
638                     readyFiller.release();
639                 }
640             }
641                    
642             WorkQueue readyQ = null;
643             Object key = readyClassQueues.poll(DEFAULT_WAIT,TimeUnit.MILLISECONDS);
644             if (key != null) {
645                 readyQ = (WorkQueue)this.allQueues.get((String)key);
646             }
647             if (readyQ != null) {
648                 while(true) { // loop left by explicit return or break on empty
649                     CrawlURI curi = null;
650                     synchronized(readyQ) {
651                         curi = readyQ.peek(this);                     
652                         if (curi != null) {
653                             // check if curi belongs in different queue
654                             String currentQueueKey = getClassKey(curi);
655                             if (currentQueueKey.equals(curi.getClassKey())) {
656                                 // curi was in right queue, emit
657                                 noteAboutToEmit(curi, readyQ);
658                                 inProcessQueues.add(readyQ);
659                                 return curi;
660                             }
661                             // URI's assigned queue has changed since it
662                             // was queued (eg because its IP has become
663                             // known). Requeue to new queue.
664                             curi.setClassKey(currentQueueKey);
665                             readyQ.dequeue(this);
666                             decrementQueuedCount(1);
667                             curi.setHolderKey(null);
668                             // curi will be requeued to true queue after lock
669                             //  on readyQ is released, to prevent deadlock
670                         } else {
671                             // readyQ is empty and ready: it's exhausted
672                             // release held status, allowing any subsequent 
673                             // enqueues to again put queue in ready
674                             readyQ.clearHeld();
675                             break;
676                         }
677                     }
678                     if(curi!=null) {
679                         // complete the requeuing begun earlier
680                         sendToQueue(curi);
681                     }
682                 }
683             } else {
684                 // ReadyQ key wasn't in all queues: unexpected
685                 if (key != null) {
686                     logger.severe("Key "+ key +
687                         " in readyClassQueues but not allQueues");
688                 }
689             }
690 
691             if(shouldTerminate) {
692                 // skip subsequent steps if already on last legs
693                 throw new EndedException("shouldTerminate is true");
694             }
695                 
696             if(inProcessQueues.size()==0) {
697                 // Nothing was ready or in progress or imminent to wake; ensure 
698                 // any piled-up pending-scheduled URIs are considered
699                 this.alreadyIncluded.requestFlush();
700             }    
701         }
702     }

724     /***
725      * Activate an inactive queue, if any are available. 
726      */
727     private void activateInactiveQueue() {
728         Object key = this.inactiveQueues.poll();
729         if (key == null) {
730             return;
731         }
732         WorkQueue candidateQ = (WorkQueue)this.allQueues.get((String)key);
733         if(candidateQ != null) {
734             synchronized(candidateQ) {
735                 replenishSessionBalance(candidateQ);
736                 if(candidateQ.isOverBudget()){
737                     // if still over-budget after an activation & replenishing,
738                     // retire
739                     retireQueue(candidateQ);
740                     return;
741                 } 
742                 long now = System.currentTimeMillis();
743                 long delay_ms = candidateQ.getWakeTime() - now;
744                 if(delay_ms>0) {
745                     // queue still due for snoozing
746                     snoozeQueue(candidateQ,now,delay_ms);
747                     return;
748                 }
749                 candidateQ.setWakeTime(0); // clear obsolete wake time, if any
750                 readyQueue(candidateQ);
751                 if (logger.isLoggable(Level.FINE)) {
752                     logger.fine("ACTIVATED queue: " +
753                         candidateQ.getClassKey());
754                    
755                 }
756             }
757         }
758     }

796     /***
797      * Wake any queues sitting in the snoozed queue whose time has come.
798      */
799     void wakeQueues() {
800         synchronized (snoozedClassQueues) {
801             long now = System.currentTimeMillis();
802             long nextWakeDelay = 0;
803             int wokenQueuesCount = 0;
804             while (true) {
805                 if (snoozedClassQueues.isEmpty()) {
806                     return;
807                 }
808                 WorkQueue peek = (WorkQueue) snoozedClassQueues.first();
809                 nextWakeDelay = peek.getWakeTime() - now;
810                 if (nextWakeDelay <= 0) {
811                     snoozedClassQueues.remove(peek);
812                     peek.setWakeTime(0);
813                     reenqueueQueue(peek);
814                     wokenQueuesCount++;
815                 } else {
816                     break;
817                 }
818             }
819             this.nextWake = new WakeTask();
820             this.wakeTimer.schedule(nextWake,nextWakeDelay);
821         }
822     }

824     /***
825      * Note that the previously emitted CrawlURI has completed
826      * its processing (for now).
827      *
828      * The CrawlURI may be scheduled to retry, if appropriate,
829      * and other related URIs may become eligible for release
830      * via the next next() call, as a result of finished().
831      *
832      *  (non-Javadoc)
833      * @see org.archive.crawler.framework.Frontier#finished(org.archive.crawler.datamodel.CrawlURI)
834      */
835     public void finished(CrawlURI curi) {
836         long now = System.currentTimeMillis();
837 
838         curi.incrementFetchAttempts();
839         logLocalizedErrors(curi);
840         WorkQueue wq = (WorkQueue) curi.getHolder();
841         assert (wq.peek(this) == curi) : "unexpected peek " + wq;
842         inProcessQueues.remove(wq, 1);
843 
844         if(includesRetireDirective(curi)) {
845             // CrawlURI is marked to trigger retirement of its queue
846             curi.processingCleanup();
847             wq.unpeek();
848             wq.update(this, curi); // rewrite any changes
849             retireQueue(wq);
850             return;
851         }
852         
853         if (needsRetrying(curi)) {
854             // Consider errors which can be retried, leaving uri atop queue
855             if(curi.getFetchStatus()!=S_DEFERRED) {
856                 wq.expend(getCost(curi)); // all retries but DEFERRED cost
857             }
858             long delay_sec = retryDelayFor(curi);
859             curi.processingCleanup(); // lose state that shouldn't burden retry
860             synchronized(wq) {
861                 wq.unpeek();
862                 // TODO: consider if this should happen automatically inside unpeek()
863                 wq.update(this, curi); // rewrite any changes
864                 if (delay_sec > 0) {
865                     long delay_ms = delay_sec * 1000;
866                     snoozeQueue(wq, now, delay_ms);
867                 } else {
868                     reenqueueQueue(wq);
869                 }
870             }
871             // Let everyone interested know that it will be retried.
872             controller.fireCrawledURINeedRetryEvent(curi);
873             doJournalRescheduled(curi);
874             return;
875         }
876 
877         // Curi will definitely be disposed of without retry, so remove from queue
878         wq.dequeue(this);
879         decrementQueuedCount(1);
880         log(curi);
881 
882         if (curi.isSuccess()) {
883             totalProcessedBytes += curi.getRecordedSize();
884             incrementSucceededFetchCount();
885             // Let everyone know in case they want to do something before we strip the curi.
886             controller.fireCrawledURISuccessfulEvent(curi);
887             doJournalFinishedSuccess(curi);
888             wq.expend(getCost(curi)); // successes cost
889         } else if (isDisregarded(curi)) {
890             // Check for codes that mean that while we the crawler did
891             // manage to schedule it, it must be disregarded for some reason.
892             incrementDisregardedUriCount();
893             // Let interested listeners know of disregard disposition.
894             controller.fireCrawledURIDisregardEvent(curi);
895             doJournalDisregarded(curi);
896             // if exception, also send to crawlErrors
897             if (curi.getFetchStatus() == S_RUNTIME_EXCEPTION) {
898                 Object[] array = { curi };
899                 controller.runtimeErrors.log(Level.WARNING, curi.getUURI()
900                         .toString(), array);
901             }
902             // TODO: consider reinstating forget-uri
903         } else {
904             // In that case FAILURE, note & log
905             //Let interested listeners know of failed disposition.
906             this.controller.fireCrawledURIFailureEvent(curi);
907             // if exception, also send to crawlErrors
908             if (curi.getFetchStatus() == S_RUNTIME_EXCEPTION) {
909                 Object[] array = { curi };
910                 this.controller.runtimeErrors.log(Level.WARNING, curi.getUURI()
911                         .toString(), array);
912             }
913             incrementFailedFetchCount();
914             // let queue note error
915             wq.noteError(((Integer) getUncheckedAttribute(curi,
916                     ATTR_ERROR_PENALTY_AMOUNT)).intValue()); 
917             doJournalFinishedFailure(curi);
918             wq.expend(getCost(curi)); // failures cost
919         }
920 
921         long delay_ms = politenessDelayFor(curi);
922         synchronized(wq) {
923             if (delay_ms > 0) {
924                 snoozeQueue(wq,now,delay_ms);
925             } else {
926                 reenqueueQueue(wq);
927             }
928         }
929 
930         curi.stripToMinimal();
931         curi.processingCleanup();
932 
933     }

939     /***
940      * Place the given queue into 'snoozed' state, ineligible to
941      * supply any URIs for crawling, for the given amount of time. 
942      * 
943      * @param wq queue to snooze 
944      * @param now time now in ms 
945      * @param delay_ms time to snooze in ms
946      */
947     private void snoozeQueue(WorkQueue wq, long now, long delay_ms) {
948         long nextTime = now + delay_ms;
949         wq.setWakeTime(nextTime);
950         long snoozeToInactiveDelayMs = ((Long)getUncheckedAttribute(null,
951                 ATTR_SNOOZE_DEACTIVATE_MS)).longValue();
952         if (delay_ms > snoozeToInactiveDelayMs && !inactiveQueues.isEmpty()) {
953             deactivateQueue(wq);
954         } else {
955             synchronized(snoozedClassQueues) {
956                 snoozedClassQueues.add(wq);
957                 if(wq == snoozedClassQueues.first()) {
958                     this.nextWake = new WakeTask();
959                     this.wakeTimer.schedule(nextWake, delay_ms);
960                 }
961             }
962         }
963     }

1266     /***
1267      * Extract some of the elements in the given collection to an
1268      * ArrayList.  This method synchronizes on the given collection's
1269      * monitor.  The returned list will never contain more than the
1270      * specified maximum number of elements.
1271      * 
1272      * @param c    the collection whose elements to extract
1273      * @param max  the maximum number of elements to extract
1274      * @return  the extraction
1275      */
1276     private static <T> ArrayList<T> extractSome(Collection<T> c, int max) {
1277         // Try to guess a sane initial capacity for ArrayList
1278         // Hopefully given collection won't grow more than 10 items
1279         // between now and the synchronized block...
