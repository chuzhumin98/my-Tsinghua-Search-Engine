37          File f = new File(absUnixPath);
38          assertTrue("Path is wrong abs " + IoUtils.getClasspathPath(f),
39              IoUtils.getClasspathPath(f).equals(absUnixPath));
40          final String relUnixPath = "one/two/three";
41          f = new File(relUnixPath);
42          assertTrue("Path is wrong rel " + IoUtils.getClasspathPath(f),
43              IoUtils.getClasspathPath(f).equals(relUnixPath));
44          final String nameUnixPath = "three";
45          f = new File(nameUnixPath);

47              IoUtils.getClasspathPath(f).equals(nameUnixPath));
